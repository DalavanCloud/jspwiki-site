070    /**
071     *  Manages plugin classes.  There exists a single instance of PluginManager
072     *  per each instance of WikiEngine, that is, each JSPWiki instance.
073     *  <P>
074     *  A plugin is defined to have three parts:
075     *  <OL>
076     *    <li>The plugin class
077     *    <li>The plugin parameters
078     *    <li>The plugin body
079     *  </ol>
080     *
081     *  For example, in the following line of code:
082     *  <pre>
083     *  [{INSERT org.apache.wiki.plugin.FunnyPlugin  foo='bar'
084     *  blob='goo'
085     *
086     *  abcdefghijklmnopqrstuvw
087     *  01234567890}]
088     *  </pre>
089     *
090     *  The plugin class is "org.apache.wiki.plugin.FunnyPlugin", the
091     *  parameters are "foo" and "blob" (having values "bar" and "goo",
092     *  respectively), and the plugin body is then
093     *  "abcdefghijklmnopqrstuvw\n01234567890".   The plugin body is
094     *  accessible via a special parameter called "_body".
095     *  <p>
096     *  If the parameter "debug" is set to "true" for the plugin,
097     *  JSPWiki will output debugging information directly to the page if there
098     *  is an exception.
099     *  <P>
100     *  The class name can be shortened, and marked without the package.
101     *  For example, "FunnyPlugin" would be expanded to
102     *  "org.apache.wiki.plugin.FunnyPlugin" automatically.  It is also
103     *  possible to define other packages, by setting the
104     *  "jspwiki.plugin.searchPath" property.  See the included
105     *  jspwiki.properties file for examples.
106     *  <P>
107     *  Even though the nominal way of writing the plugin is
108     *  <pre>
109     *  [{INSERT pluginclass WHERE param1=value1...}],
110     *  </pre>
111     *  it is possible to shorten this quite a lot, by skipping the
112     *  INSERT, and WHERE words, and dropping the package name.  For
113     *  example:
114     *
115     *  <pre>
116     *  [{INSERT org.apache.wiki.plugin.Counter WHERE name='foo'}]
117     *  </pre>
118     *
119     *  is the same as
120     *  <pre>
121     *  [{Counter name='foo'}]
122     *  </pre>
123     *  <h3>Plugin property files</h3>
124     *  <p>
125     *  Since 2.3.25 you can also define a generic plugin XML properties file per
126     *  each JAR file.
127     *  <pre>
128     *  <modules>
129     *   <plugin class="org.apache.wiki.foo.TestPlugin">
130     *       <author>Janne Jalkanen</author>
131     *       <script>foo.js</script>
132     *       <stylesheet>foo.css</stylesheet>
133     *       <alias>code</alias>
134     *   </plugin>
135     *   <plugin class="org.apache.wiki.foo.TestPlugin2">
136     *       <author>Janne Jalkanen</author>
137     *   </plugin>
138     *   </modules>
139     *  </pre>
140     *  <h3>Plugin lifecycle</h3>
141     *
142     *  <p>Plugin can implement multiple interfaces to let JSPWiki know at which stages they should
143     *  be invoked:
144     *  <ul>
145     *  <li>InitializablePlugin: If your plugin implements this interface, the initialize()-method is
146     *      called once for this class
147     *      before any actual execute() methods are called.  You should use the initialize() for e.g.
148     *      precalculating things.  But notice that this method is really called only once during the
149     *      entire WikiEngine lifetime.  The InitializablePlugin is available from 2.5.30 onwards.</li>
150     *  <li>ParserStagePlugin: If you implement this interface, the executeParse() method is called
151     *      when JSPWiki is forming the DOM tree.  You will receive an incomplete DOM tree, as well
152     *      as the regular parameters.  However, since JSPWiki caches the DOM tree to speed up later
153     *      places, which means that whatever this method returns would be irrelevant.  You can do some DOM
154     *      tree manipulation, though.  The ParserStagePlugin is available from 2.5.30 onwards.</li>
155     *  <li>WikiPlugin: The regular kind of plugin which is executed at every rendering stage.  Each
156     *      new page load is guaranteed to invoke the plugin, unlike with the ParserStagePlugins.</li>
157     *  </ul>
158     *
159     *  @since 1.6.1
160     */

180        /**
181         *  Create a new PluginManager.
182         *
183         *  @param engine WikiEngine which owns this manager.
184         *  @param props Contents of a "jspwiki.properties" file.
185         */
186        public DefaultPluginManager( WikiEngine engine, Properties props ) {
187            super( engine );
188            String packageNames = props.getProperty( PROP_SEARCHPATH );
189    
190            if( packageNames != null ) {
191                StringTokenizer tok = new StringTokenizer( packageNames, "," );
192    
193                while( tok.hasMoreTokens() ) {
194                    m_searchPath.add( tok.nextToken().trim() );
195                }
196            }
197    
198            registerPlugins();
199    
200            //
201            //  The default packages are always added.
202            //
203            m_searchPath.add( DEFAULT_PACKAGE );
204            m_searchPath.add( DEFAULT_FORMS_PACKAGE );
205    
206            PatternCompiler compiler = new Perl5Compiler();
207    
208            try {
209                m_pluginPattern = compiler.compile( PLUGIN_INSERT_PATTERN );
210            } catch( MalformedPatternException e ) {
211                log.fatal( "Internal error: someone messed with pluginmanager patterns.", e );
212                throw new InternalWikiException( "PluginManager patterns are broken" );
213            }
214    
215        }

259        /**
260         *  Outputs a HTML-formatted version of a stack trace.
261         */
262        private String stackTrace( Map< String, String > params, Throwable t ) {
263            div d = new div();
264            d.setClass( "debug" );
265            d.addElement( "Plugin execution failed, stack trace follows:" );
266            StringWriter out = new StringWriter();
267            t.printStackTrace( new PrintWriter( out ) );
268            d.addElement( new pre( out.toString() ) );
269            d.addElement( new b( "Parameters to the plugin" ) );
270    
271            ul list = new ul();
272            for( Iterator<Map.Entry< String, String > > i = params.entrySet().iterator(); i.hasNext(); ) {
273                Map.Entry< String, String > e = i.next();
274                String key = e.getKey();
275    
276                list.addElement( new li( key + "'='" + e.getValue() ) );
277            }
278    
279            d.addElement( list );
280    
281            return d.toString();
282        }

284        /**
285         *  Executes a plugin class in the given context.
286         *  <P>Used to be private, but is public since 1.9.21.
287         *
288         *  @param context The current WikiContext.
289         *  @param classname The name of the class.  Can also be a
290         *  shortened version without the package name, since the class name is searched from the
291         *  package search path.
292         *
293         *  @param params A parsed map of key-value pairs.
294         *
295         *  @return Whatever the plugin returns.
296         *
297         *  @throws PluginException If the plugin execution failed for
298         *  some reason.
299         *
300         *  @since 2.0
301         */
302        public String execute( WikiContext context, String classname, Map< String, String > params ) throws PluginException {
303            if( !m_pluginsEnabled ) {
304                return "";
305            }
306    
307            ResourceBundle rb = Preferences.getBundle( context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE );
308            boolean debug = TextUtil.isPositive( params.get( PARAM_DEBUG ) );
309            try {
310                //
311                //   Create...
312                //
313                WikiPlugin plugin = newWikiPlugin( classname, rb );
314                if( plugin == null ) {
315                    return "Plugin '" + classname + "' not compatible with this version of JSPWiki";
316                }
317    
318                //
319                //  ...and launch.
320                //
321                try {
322                    return plugin.execute( context, params );
323                } catch( PluginException e ) {
324                    if( debug ) {
325                        return stackTrace( params, e );
326                    }
327    
328                    // Just pass this exception onward.
329                    throw ( PluginException )e.fillInStackTrace();
330                } catch( Throwable t ) {
331                    // But all others get captured here.
332                    log.info( "Plugin failed while executing:", t );
333                    if( debug ) {
334                        return stackTrace( params, t );
335                    }
336    
337                    throw new PluginException( rb.getString( "plugin.error.failed" ), t );
338                }
339    
340            } catch( ClassCastException e ) {
341                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.notawikiplugin" ), classname ), e );
342            }
343        }

345        /**
346         *  Parses plugin arguments.  Handles quotes and all other kewl stuff.
347         *
348         *  <h3>Special parameters</h3>
349         *  The plugin body is put into a special parameter defined by {@link #PARAM_BODY};
350         *  the plugin's command line into a parameter defined by {@link #PARAM_CMDLINE};
351         *  and the bounds of the plugin within the wiki page text by a parameter defined
352         *  by {@link #PARAM_BOUNDS}, whose value is stored as a two-element int[] array,
353         *  i.e., <tt>[start,end]</tt>.
354         *
355         * @param argstring The argument string to the plugin.  This is
356         *  typically a list of key-value pairs, using "'" to escape
357         *  spaces in strings, followed by an empty line and then the
358         *  plugin body.  In case the parameter is null, will return an
359         *  empty parameter list.
360         *
361         * @return A parsed list of parameters.
362         *
363         * @throws IOException If the parsing fails.
364         */
365        public Map< String, String > parseArgs( String argstring ) throws IOException {
366            Map< String, String > arglist = new HashMap< String, String >();
367    
368            //
369            //  Protection against funny users.
370            //
371            if( argstring == null ) return arglist;
372    
373            arglist.put( PARAM_CMDLINE, argstring );
374    
375            StringReader    in      = new StringReader(argstring);
376            StreamTokenizer tok     = new StreamTokenizer(in);
377            int             type;
378    
379    
380            String param = null;
381            String value = null;
382    
383            tok.eolIsSignificant( true );
384    
385            boolean potentialEmptyLine = false;
386            boolean quit               = false;
387    
388            while( !quit ) {
389                String s;
390                type = tok.nextToken();
391    
392                switch( type ) {
393                  case StreamTokenizer.TT_EOF:
394                    quit = true;
395                    s = null;
396                    break;
397    
398                  case StreamTokenizer.TT_WORD:
399                    s = tok.sval;
400                    potentialEmptyLine = false;
401                    break;
402    
403                  case StreamTokenizer.TT_EOL:
404                    quit = potentialEmptyLine;
405                    potentialEmptyLine = true;
406                    s = null;
407                    break;
408    
409                  case StreamTokenizer.TT_NUMBER:
410                    s = Integer.toString( (int) tok.nval );
411                    potentialEmptyLine = false;
412                    break;
413    
414                  case '\'':
415                    s = tok.sval;
416                    break;
417    
418                  default:
419                    s = null;
420                }
421    
422                //
423                //  Assume that alternate words on the line are
424                //  parameter and value, respectively.
425                //
426                if( s != null ) {
427                    if( param == null ) {
428                        param = s;
429                    } else {
430                        value = s;
431    
432                        arglist.put( param, value );
433    
434                        // log.debug("ARG: "+param+"="+value);
435                        param = null;
436                    }
437                }
438            }
439    
440            //
441            //  Now, we'll check the body.
442            //
443            if( potentialEmptyLine ) {
444                StringWriter out = new StringWriter();
445                FileUtil.copyContents( in, out );
446    
447                String bodyContent = out.toString();
448    
449                if( bodyContent != null ) {
450                    arglist.put( PARAM_BODY, bodyContent );
451                }
452            }
453    
454            return arglist;
455        }

457        /**
458         *  Parses a plugin.  Plugin commands are of the form:
459         *  [{INSERT myplugin WHERE param1=value1, param2=value2}]
460         *  myplugin may either be a class name or a plugin alias.
461         *  <P>
462         *  This is the main entry point that is used.
463         *
464         *  @param context The current WikiContext.
465         *  @param commandline The full command line, including plugin name, parameters and body.
466         *
467         *  @return HTML as returned by the plugin, or possibly an error message.
468         *  
469         *  @throws PluginException From the plugin itself, it propagates, waah!
470         */
471        public String execute( WikiContext context, String commandline ) throws PluginException {
472            if( !m_pluginsEnabled ) {
473                return "";
474            }
475    
476            ResourceBundle rb = Preferences.getBundle( context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE );
477            PatternMatcher matcher = new Perl5Matcher();
478    
479            try {
480                if( matcher.contains( commandline, m_pluginPattern ) ) {
481                    MatchResult res = matcher.getMatch();
482    
483                    String plugin   = res.group(2);
484                    String args     = commandline.substring(res.endOffset(0),
485                                                            commandline.length() -
486                                                            (commandline.charAt(commandline.length()-1) == '}' ? 1 : 0 ) );
487                    Map<String, String> arglist  = parseArgs( args );
488    
489                    return execute( context, plugin, arglist );
490                }
491            } catch( NoSuchElementException e ) {
492                String msg =  "Missing parameter in plugin definition: "+commandline;
493                log.warn( msg, e );
494                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.missingparameter" ), commandline ) );
495            } catch( IOException e ) {
496                String msg = "Zyrf.  Problems with parsing arguments: "+commandline;
497                log.warn( msg, e );
498                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.parsingarguments" ), commandline ) );
499            }
500    
501            // FIXME: We could either return an empty string "", or
502            // the original line.  If we want unsuccessful requests
503            // to be invisible, then we should return an empty string.
504            return commandline;
505        }

507        /**
508         *  Register a plugin.
509         */
510        private void registerPlugin( WikiPluginInfo pluginClass ) {
511            String name;
512    
513            // Registrar the plugin with the className without the package-part
514            name = pluginClass.getName();
515            if( name != null ) {
516                log.debug( "Registering plugin [name]: " + name );
517                m_pluginClassMap.put( name, pluginClass );
518            }
519    
520            // Registrar the plugin with a short convenient name.
521            name = pluginClass.getAlias();
522            if( name != null ) {
523                log.debug( "Registering plugin [shortName]: " + name );
524                m_pluginClassMap.put( name, pluginClass );
525            }
526    
527            // Registrar the plugin with the className with the package-part
528            name = pluginClass.getClassName();
529            if( name != null ) {
530                log.debug( "Registering plugin [className]: " + name );
531                m_pluginClassMap.put( name, pluginClass );
532            }
533    
534            pluginClass.initializePlugin( m_engine );
535        }
536    
537        private void registerPlugins() {
538            log.info( "Registering plugins" );
539            List< Element > plugins = XmlUtil.parse( PLUGIN_RESOURCE_LOCATION, "/modules/plugin" );
540    
541            //
542            // Register all plugins which have created a resource containing its properties.
543            //
544            // Get all resources of all plugins.
545            //
546            for( Element pluginEl : plugins ) {
547                String className = pluginEl.getAttributeValue( "class" );
548                WikiPluginInfo pluginInfo = WikiPluginInfo.newInstance( className, pluginEl );
549    
550                if( pluginInfo != null ) {
551                    registerPlugin( pluginInfo );
552                }
553            }
554        }

590             *  Initializes a plugin, if it has not yet been initialized.
591             *
592             *  @param engine The WikiEngine
593             */
594            protected void initializePlugin( WikiEngine engine ) {
595                if( !m_initialized ) {
596                    // This makes sure we only try once per class, even if init fails.
597                    m_initialized = true;
598    
599                    try {
600                        WikiPlugin p = newPluginInstance();
601                        if( p instanceof InitializablePlugin ) {
602                            ( ( InitializablePlugin )p ).initialize( engine );
603                        }
604                    } catch( Exception e ) {
605                        log.info( "Cannot initialize plugin " + m_className, e );
606                    }
607                }
608            }
609

611             *  {@inheritDoc}
612             */
613            @Override
614            protected void initializeFromXML( Element el ) {
615                super.initializeFromXML( el );
616                m_alias = el.getChildText( "alias" );
617            }

626                return new WikiPluginInfo( clazz.getName() );
627            }

671            /**
672             *  Returns a text for IncludeResources.
673             *
674             *  @param type Either "script" or "stylesheet"
675             *  @return Text, or an empty string, if there is nothing to be included.
676             */
677            public String getIncludeText( String type ) {
678                try {
679                    if( "script".equals( type ) ) {
680                        return getScriptText();
681                    } else if( "stylesheet".equals( type ) ) {
682                        return getStylesheetText();
683                    }
684                } catch( Exception ex ) {
685                    // We want to fail gracefully here
686                    return ex.getMessage();
687                }
688    
689                return null;
690            }

692            private String getScriptText() throws IOException {
693                if( m_scriptText != null ) {
694                    return m_scriptText;
695                }
696    
697                if( m_scriptLocation == null ) {
698                    return "";
699                }
700    
701                try {
702                    m_scriptText = getTextResource(m_scriptLocation);
703                } catch( IOException ex ) {
704                    // Only throw this exception once!
705                    m_scriptText = "";
706                    throw ex;
707                }
708    
709                return m_scriptText;

712            private String getStylesheetText() throws IOException {
713                if( m_stylesheetText != null ) {
714                    return m_stylesheetText;
715                }
716    
717                if( m_stylesheetLocation == null ) {
718                    return "";
719                }
720    
721                try {
722                    m_stylesheetText = getTextResource(m_stylesheetLocation);
723                } catch( IOException ex ) {
724                    // Only throw this exception once!
725                    m_stylesheetText = "";
726                    throw ex;
727                }
728    
729                return m_stylesheetText;
730            }

756        /**
757         * Creates a {@link WikiPlugin}.
758         * 
759         * @param pluginName plugin's classname
760         * @param rb {@link ResourceBundle} with i18ned text for exceptions.
761         * @return a {@link WikiPlugin}.
762         * @throws PluginException if there is a problem building the {@link WikiPlugin}.
763         */
764        public WikiPlugin newWikiPlugin( String pluginName, ResourceBundle rb ) throws PluginException {
765            WikiPlugin plugin = null;
766            WikiPluginInfo pluginInfo = m_pluginClassMap.get( pluginName );
767            try {
768                if( pluginInfo == null ) {
769                    pluginInfo = WikiPluginInfo.newInstance( findPluginClass( pluginName ) );
770                    registerPlugin( pluginInfo );
771                }
772    
773                if( !checkCompatibility( pluginInfo ) ) {
774                    String msg = "Plugin '" + pluginInfo.getName() + "' not compatible with this version of JSPWiki";
775                    log.info( msg );
776                } else {
777                    plugin = pluginInfo.newPluginInstance();
778                }
779            } catch( ClassNotFoundException e ) {
780                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.couldnotfind" ), pluginName ), e );
781            } catch( InstantiationException e ) {
782                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.cannotinstantiate" ), pluginName ), e );
783            } catch( IllegalAccessException e ) {
784                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.notallowed" ), pluginName ), e );
785            } catch( Exception e ) {
786                throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.instantationfailed" ), pluginName ), e );
787            }
788            return plugin;
789        }
