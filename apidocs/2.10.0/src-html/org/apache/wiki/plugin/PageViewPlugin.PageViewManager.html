060    /**
061     * This plugin counts the number of times a page has been viewed.<br/>
062     * Parameters:
063     * <ul>
064     * <li>count=yes|no</li>
065     * <li>show=none|count|list</li>
066     * <li>entries=maximum number of list entries to be returned</li>
067     * <li>min=minimum page count to be listed</li>
068     * <li>max=maximum page count to be listed</li>
069     * <li>sort=name|count</li>
070     * </ul>
071     * Default values:<br/>
072     * <code>show=none  sort=name</code>
073     * 
074     * @since 2.8
075     */

177        /**
178         * Page view manager, handling all storage.
179         */
180        public final class PageViewManager implements WikiEventListener
181        {
182            /** Are we initialized? */
183            private boolean m_initialized = false;
184    
185            /** The page counters. */
186            private Map<String, Counter> m_counters = null;
187    
188            /** The page counters in storage format. */
189            private Properties m_storage = null;
190    
191            /** Are all changes stored? */
192            private boolean m_dirty = false;
193    
194            /** The page count storage background thread. */
195            private Thread m_pageCountSaveThread = null;
196    
197            /** The work directory. */
198            private String m_workDir = null;
199    
200            /** Comparator for descending sort on page count. */
201            private final Comparator<Object> m_compareCountDescending = new Comparator<Object>() {
202                public int compare( Object o1, Object o2 )
203                {
204                    final int v1 = getCount( o1 );
205                    final int v2 = getCount( o2 );
206                    return (v1 == v2) ? ((String) o1).compareTo( (String) o2 ) : (v1 < v2) ? 1 : -1;
207                }
208            };

210            /**
211             * Initialize the page view manager.
212             * 
213             * @param engine The wiki engine.
214             */
215            public synchronized void initialize( WikiEngine engine )
216            {
217    
218                log.info( "initializing PageView Manager" );
219    
220                m_workDir = engine.getWorkDir();
221    
222                engine.addWikiEventListener( this );
223    
224                if( m_counters == null )
225                {
226    
227                    // Load the counters into a collection
228                    m_storage = new Properties();
229                    m_counters = new TreeMap<String, Counter>();
230    
231                    loadCounters();
232                }
233    
234                // backup counters every 5 minutes
235                if( m_pageCountSaveThread == null )
236                {
237                    m_pageCountSaveThread = new CounterSaveThread( engine, 5 * STORAGE_INTERVAL, this );
238                    m_pageCountSaveThread.start();
239                }
240    
241                m_initialized = true;
242    
243            }

245            /**
246             * Handle the shutdown event via the page counter thread.
247             * 
248             */
249            private synchronized void handleShutdown()
250            {
251    
252                log.info( "handleShutdown: The counter store thread was shut down." );
253    
254                cleanup();
255    
256                if( m_counters != null )
257                {
258    
259                    m_dirty = true;
260                    storeCounters();
261    
262                    m_counters.clear();
263                    m_counters = null;
264    
265                    m_storage.clear();
266                    m_storage = null;
267                }
268    
269                m_initialized = false;
270    
271                m_pageCountSaveThread = null;
272            }

274            /**
275             * Inspect wiki events for shutdown.
276             * 
277             * @param event The wiki event to inspect.
278             */
279            public void actionPerformed( WikiEvent event )
280            {
281    
282                if( event instanceof WikiEngineEvent )
283                {
284                    if( event.getType() == WikiEngineEvent.SHUTDOWN )
285                    {
286    
287                        log.info( "Detected wiki engine shutdown" );
288                        handleShutdown();
289                    }
290                } 
291                else if( (event instanceof WikiPageRenameEvent) && (event.getType() == WikiPageRenameEvent.PAGE_RENAMED) )
292                {
293                    String oldPageName = ((WikiPageRenameEvent) event).getOldPageName();
294                    String newPageName = ((WikiPageRenameEvent) event).getNewPageName();
295                    Counter oldCounter = m_counters.get(oldPageName);
296                    if ( oldCounter != null )
297                    {
298                        m_storage.remove(oldPageName);
299                        m_counters.put(newPageName, oldCounter);
300                        m_storage.setProperty(newPageName, oldCounter.toString());
301                        m_counters.remove(oldPageName);
302                        m_dirty = true;
303                    }
304                }
305                else if( (event instanceof WikiPageEvent) && (event.getType() == WikiPageEvent.PAGE_DELETED) )
306                {
307                     String pageName = ((WikiPageEvent) event).getPageName();
308                     m_storage.remove(pageName);
309                     m_counters.remove(pageName);
310                }
311            }

313            /**
314             * Count a page hit, present a pages' counter or output a list of
315             * page counts.
316             * 
317             * @param context the wiki context
318             * @param params the plugin parameters
319             * @return String Wiki page snippet
320             * @throws PluginException Malformed pattern parameter.
321             */
322            public String execute( WikiContext context, Map<String, String> params ) throws PluginException
323            {
324                WikiEngine engine = context.getEngine();
325                WikiPage page = context.getPage();
326                String result = STR_EMPTY;
327    
328                if( page != null )
329                {
330                    // get parameters
331                    String pagename = page.getName();
332                    String count = params.get( PARAM_COUNT );
333                    String show = params.get( PARAM_SHOW );
334                    int entries = TextUtil.parseIntParameter( params.get( PARAM_MAX_ENTRIES ), Integer.MAX_VALUE );
335                    final int max = TextUtil.parseIntParameter( params.get( PARAM_MAX_COUNT ), Integer.MAX_VALUE );
336                    final int min = TextUtil.parseIntParameter( params.get( PARAM_MIN_COUNT ), Integer.MIN_VALUE );
337                    String sort = params.get( PARAM_SORT );
338                    String body = params.get( DefaultPluginManager.PARAM_BODY );
339                    Pattern[] exclude = compileGlobs( PARAM_EXCLUDE, params.get( PARAM_EXCLUDE ) );
340                    Pattern[] include = compileGlobs( PARAM_INCLUDE, params.get( PARAM_INCLUDE ) );
341                    Pattern[] refer = compileGlobs( PARAM_REFER, params.get( PARAM_REFER ) );
342                    PatternMatcher matcher = (null != exclude || null != include || null != refer) ? new Perl5Matcher() : null;
343                    boolean increment = false;
344    
345                    // increment counter?
346                    if( STR_YES.equals( count ) )
347                    {
348                        increment = true;
349                    }
350                    else
351                    {
352                        count = null;
353                    }
354    
355                    // default increment counter?
356                    if( (show == null || STR_NONE.equals( show )) && count == null )
357                    {
358                        increment = true;
359                    }
360    
361                    // filter on referring pages?
362                    Collection<String> referrers = null;
363    
364                    if( refer != null )
365                    {
366                        ReferenceManager refManager = engine.getReferenceManager();
367    
368                        Iterator iter = refManager.findCreated().iterator();
369    
370                        while ( iter != null && iter.hasNext() )
371                        {
372    
373                            String name = (String) iter.next();
374                            boolean use = false;
375    
376                            for( int n = 0; !use && n < refer.length; n++ )
377                            {
378                                use = matcher.matches( name, refer[n] );
379                            }
380    
381                            if( use )
382                            {
383                                Collection<String> refs = engine.getReferenceManager().findReferrers( name );
384    
385                                if( refs != null && !refs.isEmpty() )
386                                {
387                                    if( referrers == null )
388                                    {
389                                        referrers = new HashSet<String>();
390                                    }
391                                    referrers.addAll( refs );
392                                }
393                            }
394                        }
395                    }
396    
397                    synchronized( this )
398                    {
399                        Counter counter = m_counters.get( pagename );
400    
401                        // only count in view mode, keep storage values in sync
402                        if( increment && WikiContext.VIEW.equalsIgnoreCase( context.getRequestContext() ) )
403                        {
404                            if( counter == null )
405                            {
406                                counter = new Counter();
407                                m_counters.put( pagename, counter );
408                            }
409                            counter.increment();
410                            m_storage.setProperty( pagename, counter.toString() );
411                            m_dirty = true;
412                        }
413    
414                        if( show == null || STR_NONE.equals( show ) )
415                        {
416                            // nothing to show
417    
418                        }
419                        else if( PARAM_COUNT.equals( show ) )
420                        {
421                            // show page count
422                            if( counter == null )
423                            {
424                                counter = new Counter();
425                                m_counters.put( pagename, counter );
426                                m_storage.setProperty( pagename, counter.toString() );
427                                m_dirty = true;
428                            }
429                            result = counter.toString();
430    
431                        }
432                        else if( body != null && 0 < body.length() && STR_LIST.equals( show ) )
433                        {
434                            // show list of counts
435                            String header = STR_EMPTY;
436                            String line = body;
437                            String footer = STR_EMPTY;
438                            int start = body.indexOf( STR_SEPARATOR );
439    
440                            // split body into header, line, footer on ----
441                            // separator
442                            if( 0 < start )
443                            {
444                                header = body.substring( 0, start );
445    
446                                start = skipWhitespace( start + STR_SEPARATOR.length(), body );
447    
448                                int end = body.indexOf( STR_SEPARATOR, start );
449    
450                                if( start >= end )
451                                {
452                                    line = body.substring( start );
453    
454                                }
455                                else
456                                {
457                                    line = body.substring( start, end );
458    
459                                    end = skipWhitespace( end + STR_SEPARATOR.length(), body );
460    
461                                    footer = body.substring( end );
462                                }
463                            }
464    
465                            // sort on name or count?
466                            Map<String, Counter> sorted = m_counters;
467    
468                            if( sort != null && PARAM_COUNT.equals( sort ) )
469                            {
470                                sorted = new TreeMap<String, Counter>( m_compareCountDescending );
471    
472                                sorted.putAll( m_counters );
473                            }
474    
475                            // build a messagebuffer with the list in wiki markup
476                            StringBuffer buf = new StringBuffer( header );
477                            MessageFormat fmt = new MessageFormat( line );
478                            Object[] args = new Object[] { pagename, STR_EMPTY, STR_EMPTY };
479                            Iterator iter = sorted.entrySet().iterator();
480    
481                            while ( iter != null && 0 < entries && iter.hasNext() )
482                            {
483    
484                                Entry entry = (Entry) iter.next();
485                                String name = (String) entry.getKey();
486    
487                                // check minimum/maximum count
488                                final int value = ((Counter) entry.getValue()).getValue();
489                                boolean use = min <= value && value <= max;
490    
491                                // did we specify a refer-to page?
492                                if( use && referrers != null )
493                                {
494    
495                                    use = referrers.contains( name );
496                                }
497    
498                                // did we specify what pages to include?
499                                if( use && include != null )
500                                {
501                                    use = false;
502    
503                                    for( int n = 0; !use && n < include.length; n++ )
504                                    {
505    
506                                        use = matcher.matches( name, include[n] );
507                                    }
508                                }
509    
510                                // did we specify what pages to exclude?
511                                if( use && null != exclude )
512                                {
513                                    for( int n = 0; use && n < exclude.length; n++ )
514                                    {
515    
516                                        use &= !matcher.matches( name, exclude[n] );
517                                    }
518                                }
519    
520                                if( use )
521                                {
522                                    args[1] = engine.beautifyTitle( name );
523                                    args[2] = entry.getValue();
524    
525                                    fmt.format( args, buf, null );
526    
527                                    entries--;
528                                }
529                            }
530                            buf.append( footer );
531    
532                            // let the engine render the list
533                            result = engine.textToHTML( context, buf.toString() );
534                        }
535                    }
536                }
537                return result;
538            }

540            /**
541             * Compile regexp parameter.
542             * 
543             * @param name The name of the parameter.
544             * @param value The parameter value.
545             * @return Pattern[] The compiled patterns, or <code>null</code>.
546             * @throws PluginException On malformed patterns.
547             */
548            private Pattern[] compileGlobs( String name, String value ) throws PluginException
549            {
550    
551                Pattern[] result = null;
552    
553                if( value != null && 0 < value.length() && !STR_GLOBSTAR.equals( value ) )
554                {
555                    try
556                    {
557                        PatternCompiler pc = new GlobCompiler();
558    
559                        String[] ptrns = StringUtils.split( value, STR_COMMA );
560    
561                        result = new Pattern[ptrns.length];
562    
563                        for( int n = 0; n < ptrns.length; n++ )
564                        {
565                            result[n] = pc.compile( ptrns[n] );
566                        }
567                    }
568                    catch( MalformedPatternException e )
569                    {
570                        throw new PluginException( "Parameter " + name + " has a malformed pattern: " + e.getMessage() );
571                    }
572                }
573    
574                return result;
575            }
576    
577            /**
578             * Adjust offset skipping whitespace.
579             * 
580             * @param offset The offset in value to adjust.
581             * @param value String in which offset points.
582             * @return int Adjusted offset into value.
583             */
584            private int skipWhitespace( int offset, String value )
585            {
586                while ( Character.isWhitespace( value.charAt( offset ) ) )
587                {
588                    offset++;
589                }
590                return offset;
591            }

599            protected int getCount( Object key )

601                return m_counters.get( key ).getValue();

604            /**
605             * Load the page view counters from file.
606             */
607            private void loadCounters()
608            {
609                if( m_counters != null && m_storage != null )
610                {
611                    log.info( "Loading counters." );
612                    synchronized( this )
613                    {
614                        InputStream fis = null;
615                        try
616                        {
617                            fis = new FileInputStream( new File( m_workDir, COUNTER_PAGE ) );
618                            m_storage.load( fis );
619                        }
620                        catch( IOException ioe )
621                        {
622                            log.error( "Can't load page counter store: " + ioe.getMessage() + " , will create a new one!" );
623                        }
624                        finally
625                        {
626                            try
627                            {
628                                if( fis != null )
629                                {
630                                    fis.close();
631                                }
632                            }
633                            catch( Exception ignore )
634                            {
635                                /* ignore */
636                            }
637                        }
638    
639                        // Copy the collection into a sorted map
640                        Iterator iter = m_storage.entrySet().iterator();
641    
642                        while ( iter != null && iter.hasNext() )
643                        {
644                            Entry entry = (Entry) iter.next();
645    
646                            m_counters.put( (String) entry.getKey(), new Counter( (String) entry.getValue() ) );
647                        }
648                        
649                        log.info( "Loaded " + m_counters.size() + " counter values." );
650                    }
651                }
652            }
653    
654            /**
655             * Save the page view counters to file.
656             * 
657             */
658            protected void storeCounters()
659            {
660                if( m_counters != null && m_storage != null && m_dirty )
661                {
662                    log.info( "Storing " + m_counters.size() + " counter values." );
663    
664                    synchronized( this )
665                    {
666                        OutputStream fos = null;
667    
668                        // Write out the collection of counters
669                        try
670                        {
671                            fos = new FileOutputStream( new File( m_workDir, COUNTER_PAGE ) );
672    
673                            m_storage.store( fos, "\n# The number of times each page has been viewed.\n# Do not modify.\n" );
674                            fos.flush();
675    
676                            m_dirty = false;
677                        }
678                        catch( IOException ioe )
679                        {
680                            log.error( "Couldn't store counters values: " + ioe.getMessage() );
681                        }
682                        finally
683                        {
684                            try
685                            {
686                                if( fos != null )
687                                {
688                                    fos.close();
689                                }
690                            }
691                            catch( Exception ignore )
692                            {
693                                /* ignore */
694                            }
695                        }
696                    }
697                }
698            }

703             * @return boolean <code>true</code> if the thread is still the current
704             *         background thread.
705             * @param thrd

707            private synchronized boolean isRunning( Thread thrd )

715        /**
716         * Counter for page hits collection.
717         */
718        private static final class Counter
719        {
720    
721            /** The count value. */
722            private int m_count = 0;
723    
724            /**
725             * Create a new counter.
726             */
727            public Counter()
728            {
729            }
730    
731            /**
732             * Create and initialize a new counter.
733             * 
734             * @param value Count value.
735             */
736            public Counter( String value )
737            {
738                setValue( value );
739            }
740    
741            /**
742             * Increment counter.
743             */
744            public void increment()
745            {
746                m_count++;
747            }
748    
749            /**
750             * Get the count value.
751             * 
752             * @return int
753             */
754            public int getValue()
755            {
756                return m_count;
757            }
758    
759            /**
760             * Set the count value.
761             * 
762             * @param value String representation of the count.
763             */
764            public void setValue( String value )
765            {
766                try
767                {
768                    m_count = Integer.parseInt( value );
769    
770                }
771                catch( Exception ignore )
772                {
773                    m_count = 0;
774                }
775            }
776    
777            /**
778             * @return String String representation of the count.
779             */
780            public String toString()
781            {
782                return String.valueOf( m_count );
783            }

786        /**
787         * Background thread storing the page counters.
788         */
789        static final class CounterSaveThread extends WikiBackgroundThread
790        {
791    
792            /** The page view manager. */
793            private final PageViewManager m_manager;
794    
795            /**
796             * Create a wiki background thread to store the page counters.
797             * 
798             * @param engine The wiki engine.
799             * @param interval Delay in seconds between saves.
800             * @param pageViewManager
801             */
802            public CounterSaveThread( WikiEngine engine, int interval, PageViewManager pageViewManager )
803            {
804    
805                super( engine, interval );
806    
807                if( pageViewManager == null )
808                {
809                    throw new IllegalArgumentException( "Manager cannot be null" );
810                }
811    
812                m_manager = pageViewManager;
813            }
814    
815            /**
816             * Save the page counters to file.
817             */
818            public void backgroundTask()
