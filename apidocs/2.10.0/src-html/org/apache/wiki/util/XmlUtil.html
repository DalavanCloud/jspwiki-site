019    package org.apache.wiki.util;
020    
021    import java.io.IOException;
022    import java.io.InputStream;
023    import java.net.URL;
024    import java.util.ArrayList;
025    import java.util.Collections;
026    import java.util.Enumeration;
027    import java.util.HashSet;
028    import java.util.List;
029    import java.util.Set;
030    
031    import org.apache.commons.lang.StringUtils;
032    import org.apache.log4j.Logger;
033    import org.jdom2.Document;
034    import org.jdom2.Element;
035    import org.jdom2.JDOMException;
036    import org.jdom2.filter.Filters;
037    import org.jdom2.input.SAXBuilder;
038    import org.jdom2.xpath.XPathExpression;
039    import org.jdom2.xpath.XPathFactory;
040    
041    
042    /**
043     * Utility class to parse XML files.
044     *
045     * @since 2.10
046     */
047    public final class XmlUtil {
048        
049        private static final Logger log = Logger.getLogger( XmlUtil.class );
050        
051        private XmlUtil() {}
052        
053        /**
054         * Parses the given XML file and returns the requested nodes. If there's an error accessing or parsing the file, an
055         * empty list is returned.
056         * 
057         * @param xml file to parse; matches all resources from classpath, filters repeated items.
058         * @param requestedNodes requestd nodes on the xml file
059         * @return the requested nodes of the XML file.
060         */
061        public static List< Element > parse( String xml, String requestedNodes ) {
062            if( StringUtils.isNotEmpty( xml ) && StringUtils.isNotEmpty( requestedNodes ) ) {
063                Set< Element > readed = new HashSet< Element >();
064                SAXBuilder builder = new SAXBuilder();
065                try {
066                    Enumeration< URL > resources = XmlUtil.class.getClassLoader().getResources( xml );
067                    while( resources.hasMoreElements() ) {
068                        URL resource = resources.nextElement();
069                        
070                        log.debug( "reading " + resource.toString() );
071                        Document doc = builder.build( resource );
072                        XPathFactory xpfac = XPathFactory.instance();
073                        XPathExpression< Element > xp = xpfac.compile( requestedNodes, Filters.element() );
074                        readed.addAll( xp.evaluate( doc ) ); // filter out repeated items
075                    }
076                    return new ArrayList< Element >( readed );
077                } catch ( IOException ioe ) {
078                    log.error( "Couldn't load all " + xml + " resources", ioe );
079                } catch ( JDOMException jdome ) {
080                    log.error( "error parsing " + xml + " resources", jdome );
081                }
082            }
083            
084            return Collections.< Element >emptyList();
085        }
086        
087        /**
088         * Parses the given stream and returns the requested nodes. If there's an error accessing or parsing the stream, an
089         * empty list is returned.
090         * 
091         * @param xmlStream stream to parse.
092         * @param requestedNodes requestd nodes on the xml stream.
093         * @return the requested nodes of the XML stream.
094         */
095        public static List< Element > parse( InputStream xmlStream, String requestedNodes ) {
096            if( xmlStream != null && StringUtils.isNotEmpty( requestedNodes ) ) {
097                SAXBuilder builder = new SAXBuilder();
098                try {
099                    Document doc = builder.build( xmlStream );
100                    XPathFactory xpfac = XPathFactory.instance();
101                    XPathExpression< Element > xp = xpfac.compile( requestedNodes, Filters.element() );
102                    
103                    return xp.evaluate( doc );
104                } catch ( IOException ioe ) {
105                    log.error( "Couldn't load all " + xmlStream + " resources", ioe );
106                } catch ( JDOMException jdome ) {
107                    log.error( "error parsing " + xmlStream + " resources", jdome );
108                }
109            }
110            
111            return Collections.< Element >emptyList();
112        }
113    
114    }
