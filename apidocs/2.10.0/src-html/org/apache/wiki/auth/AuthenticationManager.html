064    /**
065     * Manages authentication activities for a WikiEngine: user login, logout, and
066     * credential refreshes. This class uses JAAS to determine how users log in.
067     * <p>
068     * The login procedure is protected in addition by a mechanism which prevents
069     * a hacker to try and force-guess passwords by slowing down attempts to log in
070     * into the same account.  Every login attempt is recorded, and stored for a while
071     * (currently ten minutes), and each login attempt during that time incurs a penalty
072     * of 2^login attempts milliseconds - that is, 10 login attempts incur a login penalty of 1.024 seconds.
073     * The delay is currently capped to 20 seconds.
074     * 
075     * @since 2.3
076     */
077    public class AuthenticationManager {

168        /**
169         * Creates an AuthenticationManager instance for the given WikiEngine and
170         * the specified set of properties. All initialization for the modules is
171         * done here.
172         * @param engine the wiki engine
173         * @param props the properties used to initialize the wiki engine
174         * @throws WikiException if the AuthenticationManager cannot be initialized

176        @SuppressWarnings("unchecked")
177        public void initialize( WikiEngine engine, Properties props ) throws WikiException
178        {
179            m_engine = engine;
180            m_storeIPAddress = TextUtil.getBooleanProperty( props, PROP_STOREIPADDRESS, m_storeIPAddress );
181    
182            // Should J2SE policies be used for authorization?
183            m_useJAAS = SECURITY_JAAS.equals(props.getProperty( PROP_SECURITY, SECURITY_JAAS ));
184            
185            // Should we allow cookies for assertions? (default: yes)
186            m_allowsCookieAssertions = TextUtil.getBooleanProperty( props,
187                                                                  PROP_ALLOW_COOKIE_ASSERTIONS,
188                                                                  true );
189            
190            // Should we allow cookies for authentication? (default: no)
191            m_allowsCookieAuthentication = TextUtil.getBooleanProperty( props,
192                                                                        PROP_ALLOW_COOKIE_AUTH,
193                                                                        false );
194            
195            // Should we throttle logins? (default: yes)
196            m_throttleLogins = TextUtil.getBooleanProperty( props,
197                                                            PROP_LOGIN_THROTTLING,
198                                                            true );
199    
200            // Look up the LoginModule class
201            String loginModuleClassName = TextUtil.getStringProperty( props, PROP_LOGIN_MODULE, DEFAULT_LOGIN_MODULE );
202            try
203            {
204                m_loginModuleClass = (Class<? extends LoginModule>) Class.forName( loginModuleClassName );
205            }
206            catch (ClassNotFoundException e)
207            {
208                e.printStackTrace();
209                throw new WikiException( "Could not instantiate LoginModule class.", e );
210            }
211            
212            // Initialize the LoginModule options
213            initLoginModuleOptions( props );
214        }

240            }

244        /**
245         * <p>Logs in the user by attempting to populate a WikiSession Subject from
246         * a web servlet request by examining the request
247         *  for the presence of container credentials and user cookies. The processing
248         * logic is as follows:
249         * </p>
250         * <ul>
251         * <li>If the WikiSession had previously been unauthenticated, check to see if
252         * user has subsequently authenticated. To be considered "authenticated,"
253         * the request must supply one of the following (in order of preference):
254         * the container <code>userPrincipal</code>, container <code>remoteUser</code>,
255         * or authentication cookie. If the user is authenticated, this method fires event
256         * {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_AUTHENTICATED}
257         * with two parameters: a Principal representing the login principal,
258         * and the current WikiSession. In addition, if the authorizer is of type
259         * WebContainerAuthorizer, this method iterates through the container roles returned by
260         * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer#getRoles()},
261         * tests for membership in each one, and adds those that pass to the Subject's principal set.</li>
262         * <li>If, after checking for authentication, the WikiSession is still Anonymous,
263         * this method next checks to see if the user has "asserted" an identity
264         * by supplying an assertion cookie. If the user is found to be asserted,
265         * this method fires event {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_ASSERTED}
266         * with two parameters: <code>WikiPrincipal(<em>cookievalue</em>)</code>, and
267         * the current WikiSession.</li>
268         * <li>If, after checking for authenticated and asserted status, the  WikiSession is
269         * <em>still</em> anonymous, this method fires event
270         * {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_ANONYMOUS} with
271         * two parameters: <code>WikiPrincipal(<em>remoteAddress</em>)</code>,
272         * and the current WikiSession </li>
273         * </ul>
274         * @param request servlet request for this user
275         * @return always returns <code>true</code> (because anonymous login, at least, will always succeed)
276         * @throws org.apache.wiki.auth.WikiSecurityException if the user cannot be logged in for any reason
277         * @since 2.3
278         */
279        public boolean login( HttpServletRequest request ) throws WikiSecurityException
280        {
281            HttpSession httpSession = request.getSession();
282            WikiSession session = SessionMonitor.getInstance(m_engine).find( httpSession );
283            AuthenticationManager authenticationMgr = m_engine.getAuthenticationManager();
284            AuthorizationManager authorizationMgr = m_engine.getAuthorizationManager();
285            CallbackHandler handler = null;
286            Map<String,String> options = EMPTY_MAP;
287    
288            // If user not authenticated, check if container logged them in, or if
289            // there's an authentication cookie
290            if ( !session.isAuthenticated() )
291            {
292                // Create a callback handler
293                handler = new WebContainerCallbackHandler( m_engine, request );
294                
295                // Execute the container login module, then (if that fails) the cookie auth module
296                Set<Principal> principals = authenticationMgr.doJAASLogin( WebContainerLoginModule.class, handler, options );
297                if ( principals.size() == 0 && authenticationMgr.allowsCookieAuthentication() )
298                {
299                    principals = authenticationMgr.doJAASLogin( CookieAuthenticationLoginModule.class, handler, options );
300                }
301                
302                // If the container logged the user in successfully, tell the WikiSession (and add all of the Principals)
303                if ( principals.size() > 0 )
304                {
305                    fireEvent( WikiSecurityEvent.LOGIN_AUTHENTICATED, getLoginPrincipal( principals ), session );
306                    for ( Principal principal : principals )
307                    {
308                        fireEvent( WikiSecurityEvent.PRINCIPAL_ADD, principal, session );
309                    }
310                    
311                    // Add all appropriate Authorizer roles
312                    injectAuthorizerRoles( session, authorizationMgr.getAuthorizer(), request );
313                }
314            }
315    
316            // If user still not authenticated, check if assertion cookie was supplied
317            if ( !session.isAuthenticated() && authenticationMgr.allowsCookieAssertions() )
318            {
319                // Execute the cookie assertion login module
320                Set<Principal> principals = authenticationMgr.doJAASLogin( CookieAssertionLoginModule.class, handler, options );
321                if ( principals.size() > 0 )
322                {
323                    fireEvent( WikiSecurityEvent.LOGIN_ASSERTED, getLoginPrincipal( principals ), session);
324                }
325            }
326    
327            // If user still anonymous, use the remote address
328            if (session.isAnonymous() )
329            {
330                Set<Principal> principals = authenticationMgr.doJAASLogin( AnonymousLoginModule.class, handler, options );
331                if ( principals.size() > 0 )
332                {
333                    fireEvent( WikiSecurityEvent.LOGIN_ANONYMOUS, getLoginPrincipal( principals ), session );
334                    return true;
335                }
336            }
337            
338            // If by some unusual turn of events the Anonymous login module doesn't work, login failed!
339            return false;
340        }

354         * @deprecated use {@link #login(WikiSession, HttpServletRequest, String, String)} instead

361        /**
362         * Attempts to perform a WikiSession login for the given username/password
363         * combination using JSPWiki's custom authentication mode. In order to log in,
364         * the JAAS LoginModule supplied by the WikiEngine property {@link #PROP_LOGIN_MODULE}
365         * will be instantiated, and its
366         * {@link javax.security.auth.spi.LoginModule#initialize(Subject, CallbackHandler, Map, Map)}
367         * method will be invoked. By default, the {@link org.apache.wiki.auth.login.UserDatabaseLoginModule}
368         * class will be used. When the LoginModule's <code>initialize</code> method is invoked,
369         * an options Map populated by properties keys prefixed by {@link #PREFIX_LOGIN_MODULE_OPTIONS}
370         * will be passed as a parameter.
371         * @param session the current wiki session; may not be <code>null</code>.
372         * @param request the user's HTTP request. This parameter may be <code>null</code>, but the configured
373         * LoginModule will not have access to the HTTP request in this case.
374         * @param username The user name. This is a login name, not a WikiName. In
375         *            most cases they are the same, but in some cases, they might
376         *            not be.
377         * @param password the password
378         * @return true, if the username/password is valid
379         * @throws org.apache.wiki.auth.WikiSecurityException if the Authorizer or UserManager cannot be obtained
380         */
381        public boolean login( WikiSession session, HttpServletRequest request, String username, String password ) throws WikiSecurityException
382        {
383            if ( session == null )
384            {
385                log.error( "No wiki session provided, cannot log in." );
386                return false;
387            }
388    
389            // Protect against brute-force password guessing if configured to do so
390            if ( m_throttleLogins )
391            {
392                delayLogin(username);
393            }
394            
395            CallbackHandler handler = new WikiCallbackHandler(
396                    m_engine,
397                    null,
398                    username,
399                    password );
400            
401            // Execute the user's specified login module
402            Set<Principal> principals = doJAASLogin( m_loginModuleClass, handler, m_loginModuleOptions );
403            if (principals.size() > 0)
404            {
405                fireEvent(WikiSecurityEvent.LOGIN_AUTHENTICATED, getLoginPrincipal( principals ), session );
406                for ( Principal principal : principals )
407                {
408                    fireEvent( WikiSecurityEvent.PRINCIPAL_ADD, principal, session );
409                }
410                
411                // Add all appropriate Authorizer roles
412                injectAuthorizerRoles( session, m_engine.getAuthorizationManager().getAuthorizer(), null );
413                
414                return true;
415            }
416            return false;
417        }

419        /**
420         *  This method builds a database of login names that are being attempted, and will try to
421         *  delay if there are too many requests coming in for the same username.
422         *  <p>
423         *  The current algorithm uses 2^loginattempts as the delay in milliseconds, i.e.
424         *  at 10 login attempts it'll add 1.024 seconds to the login.
425         *  
426         *  @param username The username that is being logged in
427         */
428        private void delayLogin( String username )
429        {
430            try
431            {
432                m_lastLoginAttempts.cleanup( LASTLOGINS_CLEANUP_TIME );
433                int count = m_lastLoginAttempts.count( username );
434                
435                long delay = Math.min( 1<<count, MAX_LOGIN_DELAY );
436                log.debug( "Sleeping for "+delay+" ms to allow login." );
437                Thread.sleep( delay );
438                
439                m_lastLoginAttempts.add( username );
440            }
441            catch( InterruptedException e )
442            {
443                // FALLTHROUGH is fine
444            }
445        }

447        /**
448         * Logs the user out by retrieving the WikiSession associated with the
449         * HttpServletRequest and unbinding all of the Subject's Principals,
450         * except for {@link Role#ALL}, {@link Role#ANONYMOUS}.
451         * is a cheap-and-cheerful way to do it without invoking JAAS LoginModules.
452         * The logout operation will also flush the JSESSIONID cookie from
453         * the user's browser session, if it was set.
454         * @param request the current HTTP request
455         */
456        public void logout( HttpServletRequest request )
457        {
458            if( request == null )
459            {
460                log.error( "No HTTP reqest provided; cannot log out." );
461                return;
462            }
463    
464            HttpSession session = request.getSession();
465            String sid = ( session == null ) ? "(null)" : session.getId();
466            if( log.isDebugEnabled() )
467            {
468                log.debug( "Invalidating WikiSession for session ID=" + sid );
469            }
470            // Retrieve the associated WikiSession and clear the Principal set
471            WikiSession wikiSession = WikiSession.getWikiSession( m_engine, request );
472            Principal originalPrincipal = wikiSession.getLoginPrincipal();
473            wikiSession.invalidate();
474    
475            // Remove the wikiSession from the WikiSession cache
476            WikiSession.removeWikiSession( m_engine, request );
477    
478            // We need to flush the HTTP session too
479            if ( session != null )
480            {
481                session.invalidate();
482            }
483    
484            // Log the event
485            fireEvent( WikiSecurityEvent.LOGOUT, originalPrincipal, null );
486        }

537        /**
538         * Instantiates and executes a single JAAS
539         * {@link javax.security.auth.spi.LoginModule}, and returns a Set of
540         * Principals that results from a successful login. The LoginModule is instantiated,
541         * then its {@link javax.security.auth.spi.LoginModule#initialize(Subject, CallbackHandler, Map, Map)}
542         * method is called. The parameters passed to <code>initialize</code> is a 
543         * dummy Subject, an empty shared-state Map, and an options Map the caller supplies.
544         * 
545         * @param clazz
546         *            the LoginModule class to instantiate
547         * @param handler
548         *            the callback handler to supply to the LoginModule
549         * @param options
550         *            a Map of key/value strings for initializing the LoginModule
551         * @return the set of Principals returned by the JAAS method {@link Subject#getPrincipals()}
552         * @throws WikiSecurityException
553         *             if the LoginModule could not be instantiated for any reason
554         */
555        protected Set<Principal> doJAASLogin(Class<? extends LoginModule> clazz, CallbackHandler handler, Map<String,String> options) throws WikiSecurityException
556        {
557            // Instantiate the login module
558            LoginModule loginModule = null;
559            try
560            {
561                loginModule = clazz.newInstance();
562            }
563            catch (InstantiationException e)
564            {
565                throw new WikiSecurityException(e.getMessage(), e );
566            }
567            catch (IllegalAccessException e)
568            {
569                throw new WikiSecurityException(e.getMessage(), e );
570            }
571    
572            // Initialize the LoginModule
573            Subject subject = new Subject();
574            loginModule.initialize( subject, handler, EMPTY_MAP, options );
575    
576            // Try to log in:
577            boolean loginSucceeded = false;
578            boolean commitSucceeded = false;
579            try
580            {
581                loginSucceeded = loginModule.login();
582                if (loginSucceeded)
583                {
584                    commitSucceeded = loginModule.commit();
585                }
586            }
587            catch (LoginException e)
588            {
589                // Login or commit failed! No principal for you!
590            }
591    
592            // If we successfully logged in & committed, return all the principals
593            if (loginSucceeded && commitSucceeded)
594            {
595                return subject.getPrincipals();
596            }
597            return NO_PRINCIPALS;
598        }

600        /**
601         * Looks up and obtains a configuration file inside the WEB-INF folder of a
602         * wiki webapp.
603         * @param engine the wiki engine

605         * @return the URL to the file
606         */
607        protected static URL findConfigFile( WikiEngine engine, String name )
608        {
609            // Try creating an absolute path first
610            File defaultFile = null;
611            if( engine.getRootPath() != null )
612            {
613                defaultFile = new File( engine.getRootPath() + "/WEB-INF/" + name );
614            }
615            if ( defaultFile != null && defaultFile.exists() )
616            {
617                try
618                {
619                    return defaultFile.toURI().toURL();
620                }
621                catch ( MalformedURLException e)
622                {
623                    // Shouldn't happen, but log it if it does
624                    log.warn( "Malformed URL: " + e.getMessage() );
625                }
626    
627            }
628    
629            // Ok, the absolute path didn't work; try other methods
630    
631            URL path = null;
632            
633            if( engine.getServletContext() != null )
634            {
635                OutputStream os = null;
636                InputStream is = null;
637                try
638                {
639                    //  create a tmp file of the policy loaded as an InputStream and return the URL to it
640                    //  
641                    is = AuthenticationManager.class.getResourceAsStream( "/" + name );
642                    File tmpFile = File.createTempFile( "temp." + name, "" );
643                    tmpFile.deleteOnExit();
644    
645                    os = new FileOutputStream(tmpFile);
646    
647                    byte[] buff = new byte[1024];
648    
649                    while( is.read(buff) != -1 )
650                    {
651                        os.write(buff);
652                    }
653    
654                    path = tmpFile.toURI().toURL();
655                }
656                catch( MalformedURLException e )
657                {
658                    // This should never happen unless I screw up
659                    log.fatal( "Your code is b0rked.  You are a bad person.", e );
660                }
661                catch (IOException e)
662                {
663                   log.error( "failed to load security policy from file " + name + ",stacktrace follows", e );
664                }
665                finally 
666                {
667                    IOUtils.closeQuietly( is );
668                    IOUtils.closeQuietly( os );
669                }
670            }
671            return path;
672        }

731        /**
732         * Initializes the options Map supplied to the configured LoginModule every time it is invoked.
733         * The properties and values extracted from
734         * <code>jspwiki.properties</code> are of the form
735         * <code>jspwiki.loginModule.options.<var>param</var> = <var>value</var>, where
736         * <var>param</var> is the key name, and <var>value</var> is the value.
737         * @param props the properties used to initialize JSPWiki
738         * @throws IllegalArgumentException if any of the keys are duplicated
739         */
740        private void initLoginModuleOptions(Properties props)
741        {
742            for ( Object key : props.keySet() )
743            {
744                String propName = key.toString();
745                if ( propName.startsWith( PREFIX_LOGIN_MODULE_OPTIONS ) )
746                {
747                    // Extract the option name and value
748                    String optionKey = propName.substring( PREFIX_LOGIN_MODULE_OPTIONS.length() ).trim();
749                    if ( optionKey.length() > 0 )
750                    {
751                        String optionValue = props.getProperty( propName );
752                        
753                        // Make sure the key is unique before stashing the key/value pair
754                        if ( m_loginModuleOptions.containsKey( optionKey ) )
755                        {
756                            throw new IllegalArgumentException( "JAAS LoginModule key " + propName + " cannot be specified twice!" );
757                        }
758                        m_loginModuleOptions.put( optionKey, optionValue );
759                    }
760                }
761            }
762        }

764        /**
765         * After successful login, this method is called to inject authorized role Principals into the WikiSession.
766         * To determine which roles should be injected, the configured Authorizer
767         * is queried for the roles it knows about by calling  {@link org.apache.wiki.auth.Authorizer#getRoles()}.
768         * Then, each role returned by the authorizer is tested by calling {@link org.apache.wiki.auth.Authorizer#isUserInRole(WikiSession, Principal)}.
769         * If this check fails, and the Authorizer is of type WebAuthorizer, the role is checked again by calling
770         * {@link org.apache.wiki.auth.authorize.WebAuthorizer#isUserInRole(javax.servlet.http.HttpServletRequest, Principal)}).
771         * Any roles that pass the test are injected into the Subject by firing appropriate authentication events.
772         * @param session the user's current WikiSession
773         * @param authorizer the WikiEngine's configured Authorizer
774         * @param request the user's HTTP session, which may be <code>null</code>
775         */
776        private void injectAuthorizerRoles( WikiSession session, Authorizer authorizer, HttpServletRequest request )
777        {
778            // Test each role the authorizer knows about
779            for ( Principal role : authorizer.getRoles() )
780            {
781                // Test the Authorizer
782                if ( authorizer.isUserInRole( session, role ) )
783                {
784                    fireEvent( WikiSecurityEvent.PRINCIPAL_ADD, role, session );
785                    if ( log.isDebugEnabled() )
786                    {
787                        log.debug("Added authorizer role " + role.getName() + "." );
788                    }
789                }
790                
791                // If web authorizer, test the request.isInRole() method also
792                else if ( request != null && authorizer instanceof WebAuthorizer )
793                {
794                    WebAuthorizer wa = (WebAuthorizer)authorizer;
795                    if ( wa.isUserInRole( request, role ) )
796                    {
797                        fireEvent( WikiSecurityEvent.PRINCIPAL_ADD, role, session );
798                        if ( log.isDebugEnabled() )
799                        {
800                            log.debug("Added container role " + role.getName() + "." );
801                        }
802                    }
803                }
804            }
805        }
