045    /**
046     * Manages JSPWiki installation on behalf of <code>admin/Install.jsp</code>.
047     * The contents of this class were previously part of <code>Install.jsp</code>.
048     *
049     * @since 2.4.20
050     */
051    public class Installer
052    {
053        public static final String ADMIN_ID = "admin";
054        public static final String ADMIN_NAME = "Administrator";
055        public static final String INSTALL_INFO = "Installer.Info";
056        public static final String INSTALL_ERROR = "Installer.Error";
057        public static final String INSTALL_WARNING = "Installer.Warning";
058        public static final String APP_NAME = WikiEngine.PROP_APPNAME;
059        public static final String BASE_URL = WikiEngine.PROP_BASEURL;
060        public static final String STORAGE_DIR = BasicAttachmentProvider.PROP_STORAGEDIR;
061        public static final String LOG_DIR = "log4j.appender.FileLog.File";
062        public static final String PAGE_DIR = FileSystemProvider.PROP_PAGEDIR;
063        public static final String WORK_DIR = WikiEngine.PROP_WORKDIR;
064        public static final String ADMIN_GROUP = "Admin";
065        private final WikiSession m_session;
066        private final File m_propertyFile;
067        private final Properties m_props;
068        private final WikiEngine m_engine;
069        private HttpServletRequest m_request;
070        private boolean m_validated;
071        
072        public Installer( HttpServletRequest request, ServletConfig config )
073        {
074            // Get wiki session for this user
075            m_engine = WikiEngine.getInstance( config );
076            m_session = WikiSession.getWikiSession( m_engine, request );
077            
078            // Get the servlet context, and file for properties
079            ServletContext context = config.getServletContext();
080            String path = context.getRealPath("/");
081            m_propertyFile = new File( path, PropertyReader.DEFAULT_JSPWIKI_CONFIG );
082            m_props = new CommentedProperties();
083            
084            // Stash the request
085            m_request = request;
086            m_validated = false;
087        }
088        
089        /**
090         * Returns <code>true</code> if the administrative user had
091         * been created previously.
092         * @return the result
093         */
094        public boolean adminExists()
095        {
096            // See if the admin user exists already
097            UserManager userMgr = m_engine.getUserManager();
098            UserDatabase userDb = userMgr.getUserDatabase();
099            
100            try
101            {
102                userDb.findByLoginName( ADMIN_ID );
103                return true;
104            }
105            catch ( NoSuchPrincipalException e )
106            {
107                return false;
108            }
109        }
110        
111        /**
112         * Creates an adminstrative user and returns the new password.
113         * If the admin user exists, the password will be <code>null</code>.
114         * @return the password
115         * @throws WikiSecurityException
116         */
117        public String createAdministrator() throws WikiSecurityException
118        {
119            if ( !m_validated )
120            {
121                throw new WikiSecurityException( "Cannot create administrator because one or more of the installation settings are invalid." );
122            }
123            
124            if ( adminExists() )
125            {
126                return null;
127            }
128            
129            // See if the admin user exists already
130            UserManager userMgr = m_engine.getUserManager();
131            UserDatabase userDb = userMgr.getUserDatabase();
132            String password = null;
133            
134            try
135            {
136                userDb.findByLoginName( ADMIN_ID );
137            }
138            catch ( NoSuchPrincipalException e )
139            {
140                // Create a random 12-character password
141                password = TextUtil.generateRandomPassword();
142                UserProfile profile = userDb.newProfile();
143                profile.setLoginName( ADMIN_ID );
144                profile.setFullname( ADMIN_NAME );
145                profile.setPassword( password );
146                userDb.save( profile );
147            }
148            
149            // Create a new admin group
150            GroupManager groupMgr = m_engine.getGroupManager();
151            Group group = null;
152            try
153            {
154                group = groupMgr.getGroup( ADMIN_GROUP );
155                group.add( new WikiPrincipal( ADMIN_NAME ) );
156            }
157            catch ( NoSuchPrincipalException e )
158            {
159                group = groupMgr.parseGroup( ADMIN_GROUP, ADMIN_NAME, true );
160            }
161            groupMgr.setGroup( m_session, group );
162            
163            return password;
164        }

190        @SuppressWarnings("deprecation")
191        public void parseProperties () throws Exception
192        {
193            ResourceBundle rb = ResourceBundle.getBundle( InternationalizationManager.CORE_BUNDLE,
194                                                          m_session.getLocale() );
195            m_validated = false;
196            
197            try
198            {
199                InputStream in = null; 
200                try
201                {
202                    // Load old properties from disk
203                    in = new FileInputStream( m_propertyFile ); 
204                    m_props.load( in );
205                }
206                finally
207                {
208                    if( in != null ) 
209                    {
210                        in.close();
211                    }
212                }
213            }
214            catch( IOException e )
215            {
216                Object[] args = { e.getMessage() };
217                m_session.addMessage( INSTALL_ERROR, MessageFormat.format( 
218                                            rb.getString( "install.installer.unable.read.props" ), args ) );
219            }
220            
221            // Get application name
222            String nullValue = m_props.getProperty( APP_NAME, rb.getString( "install.installer.default.appname" ) );
223            parseProperty( APP_NAME, nullValue );
224            
225            // Get/sanitize base URL
226            nullValue = m_request.getRequestURL().toString();
227            nullValue = nullValue.substring( 0, nullValue.lastIndexOf('/') )+"/";
228            parseProperty( BASE_URL, nullValue );
229            sanitizeURL( BASE_URL );
230            
231            // Get/sanitize page directory
232            nullValue = m_props.getProperty( PAGE_DIR, rb.getString( "install.installer.default.pagedir" ) );
233            parseProperty( PAGE_DIR, nullValue );
234            sanitizePath( PAGE_DIR );
235            
236            // Get/sanitize log directory
237            nullValue = m_props.getProperty( LOG_DIR, "/tmp/" );
238            parseProperty( LOG_DIR, nullValue );
239            sanitizePath( LOG_DIR );
240            
241            // Get/sanitize work directory
242            nullValue = m_props.getProperty( WORK_DIR, "/tmp/" );
243            parseProperty( WORK_DIR, nullValue );
244            sanitizePath( WORK_DIR );
245            
246            // Get/sanitize security property
247            nullValue = m_props.getProperty( AuthenticationManager.PROP_SECURITY, AuthenticationManager.SECURITY_JAAS );
248            parseProperty( AuthenticationManager.PROP_SECURITY, nullValue );
249            
250            // Set a few more default properties, for easy setup
251            m_props.setProperty( STORAGE_DIR, m_props.getProperty( PAGE_DIR ) );
252            m_props.setProperty( PageManager.PROP_PAGEPROVIDER, "VersioningFileProvider" );
253            m_props.setProperty( WikiEngine.PROP_ENCODING, "UTF-8" );
254        }
255        
256        public void saveProperties()
257        {
258            ResourceBundle rb = ResourceBundle.getBundle( InternationalizationManager.CORE_BUNDLE,
259                                                          m_session.getLocale() );
260            // Write the file back to disk
261            try
262            {
263                OutputStream out = null;
264                try
265                {
266                    out = new FileOutputStream( m_propertyFile );
267                    m_props.store( out, null );
268                }
269                finally
270                {
271                    if ( out != null )
272                    {
273                        out.close();
274                    }
275                }
276                m_session.addMessage( INSTALL_INFO,
277                    rb.getString( "install.installer.props.saved" ) );
278            }
279            catch( IOException e )
280            {
281                Object[] args = { e.getMessage(), m_props.toString() };
282                m_session.addMessage( INSTALL_ERROR, 
283                                      MessageFormat.format( rb.getString( "install.installer.props.notsaved" ), args ) );
284            }
285        }
286        
287        public boolean validateProperties() throws Exception
288        {
289            ResourceBundle rb = ResourceBundle.getBundle( InternationalizationManager.CORE_BUNDLE,
290                                                          m_session.getLocale() );
291            m_session.clearMessages( INSTALL_ERROR );
292            parseProperties();
293            validateNotNull( BASE_URL, rb.getString( "install.installer.validate.baseurl" ) );
294            validateNotNull( PAGE_DIR, rb.getString( "install.installer.validate.pagedir" ) );
295            validateNotNull( APP_NAME, rb.getString( "install.installer.validate.appname" ) );
296            validateNotNull( WORK_DIR, rb.getString( "install.installer.validate.workdir" ) );
297            validateNotNull( LOG_DIR, rb.getString( "install.installer.validate.logdir" ) );
298            
299            if ( m_session.getMessages( INSTALL_ERROR ).length == 0 )
300            {
301                m_validated = true;
302            }
303            return m_validated;
304        }
305            
306        /**
307         * Sets a property based on the value of an HTTP request parameter.
308         * If the parameter is not found, a default value is used instead.
309         * @param param the parameter containing the value we will extract
310         * @param defaultValue the default to use if the parameter was not passed
311         * in the request
312         */
313        private void parseProperty( String param, String defaultValue )
314        {
315            String value = m_request.getParameter( param );
316            if ( value == null )
317            {
318                value = defaultValue;
319            }
320            m_props.put( param, value );
321        }
322        
323        /**
324         * Simply sanitizes any path which contains backslashes (sometimes Windows
325         * users may have them) by expanding them to double-backslashes
326         * @param key the key of the property to sanitize
327         */
328        private void sanitizePath( String key )
329        {
330            String s = m_props.getProperty( key );
331            s = TextUtil.replaceString(s, "\\", "\\\\" );
332            s = s.trim();
333            m_props.put( key, s );
334        }
335        
336        /**
337         * Simply sanitizes any URL which contains backslashes (sometimes Windows
338         * users may have them)
339         * @param key the key of the property to sanitize
340         */
341        private void sanitizeURL( String key )
342        {
343            String s = m_props.getProperty( key );
344            s = TextUtil.replaceString( s, "\\", "/" );
345            s = s.trim();
346            if (!s.endsWith("/"))
347            {
348                s = s + "/";
349            }
350            m_props.put( key, s );
351        }
352    
353        private void validateNotNull( String key, String message )
354        {
355            String value = m_props.getProperty( key );
356            if ( value == null || value.length() == 0 )
357            {
358                m_session.addMessage( INSTALL_ERROR, message );
359            }
360        }
