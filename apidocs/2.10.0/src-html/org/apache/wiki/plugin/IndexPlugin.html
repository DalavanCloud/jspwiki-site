019    package org.apache.wiki.plugin;
020    
021    
022    import java.util.*;
023    import java.util.regex.Pattern;
024    
025    import org.apache.ecs.Element;
026    import org.apache.ecs.xhtml.div;
027    import org.apache.ecs.xhtml.span;
028    import org.apache.log4j.Logger;
029    import org.apache.wiki.WikiContext;
030    import org.apache.wiki.api.exceptions.PluginException;
031    import org.apache.wiki.api.exceptions.ProviderException;
032    import org.apache.wiki.api.plugin.WikiPlugin;
033    
034    /**
035     *  A plugin that creates an index of pages according to a certain  pattern.
036     *  <br>The default is to include all pages.
037     *  <p>
038     *  This is a complete rewrite of the old IndexPlugin under an Apache license.
039     *  <p>Parameters (From AbstractReferralPlugin) : </p>
040     *  <ul>
041     *    <li><b>include</b> - A regexp pattern for marking which pages should be included.</li>
042     *    <li><b>exclude</b> - A regexp pattern for marking which pages should be excluded.</li>
043     *  </ul>
044     */
045    public class IndexPlugin  extends AbstractReferralPlugin implements WikiPlugin
046    {
047        private static Logger log = Logger.getLogger( IndexPlugin.class );
048        
049        /**
050         *  {@inheritDoc}
051         */
052        public String execute( WikiContext context, Map<String, String> params ) throws PluginException
053        {
054            String include = params.get( PARAM_INCLUDE );
055            String exclude = params.get( PARAM_EXCLUDE );
056            
057            List<String> pages;
058            div masterDiv = new div();
059            masterDiv.setClass( "index" );
060            
061            div indexDiv = new div();
062            
063            masterDiv.addElement( indexDiv );
064            indexDiv.setClass( "header" );
065            try
066            {
067                pages = listPages( context, include, exclude );
068                context.getEngine().getPageSorter().sort( pages );
069                
070                char initialChar = ' ';
071                
072                div currentDiv = new div();
073                
074                for( String name : pages )
075                {
076                    if( name.charAt( 0 ) != initialChar )
077                    {
078                        if( initialChar != ' ' ) indexDiv.addElement( " - " );
079                        initialChar = name.charAt( 0 );
080                        
081                        masterDiv.addElement( makeHeader(initialChar) );
082                
083                        currentDiv = new div();
084                        currentDiv.setClass("body");
085                        masterDiv.addElement( currentDiv );
086                        
087                        indexDiv.addElement( "<a href='#"+initialChar+"'>"+initialChar+"</a>" );
088                    }
089                    else
090                    {
091                        currentDiv.addElement( ", " );
092                    }
093                    
094                    String link = "<a href='"+
095                                  context.getURL( WikiContext.VIEW, name )+
096                                  "'>"+name+"</a>";
097                    
098                    currentDiv.addElement( link );
099                }
100            }
101            catch( ProviderException e )
102            {
103                log.warn("Could not load page index",e);
104                throw new PluginException( e.getMessage() );
105            }
106            
107            return masterDiv.toString();
108        }
109        
110        /**
111         *  Create the DOM for a heading
112         * @param initialChar
113         * @return A span element.
114         */
115        private Element makeHeader( char initialChar )
116        {
117            span s = new span();
118            s.setClass( "section" );
119            s.addElement( "<a name='"+initialChar+"'>"+initialChar+"</a>" );
120    
121            return s;
122        }
123    
124        /**
125         *  Grabs a list of all pages and filters them according to the include/exclude patterns.
126         *  
127         * @param context
128         * @param include
129         * @param exclude
130         * @return A list containing page names which matched the filters.
131         * @throws ProviderException
132         */
133        private List<String> listPages( WikiContext context, String include, String exclude )
134            throws ProviderException
135        {
136            Pattern includePtrn = include != null ? Pattern.compile( include ) : Pattern.compile(".*");
137            Pattern excludePtrn = exclude != null ? Pattern.compile( exclude ) : Pattern.compile("\\p{Cntrl}"); // There are no control characters in page names
138            
139            ArrayList<String> result = new ArrayList<String>();
140            
141            Collection pages = context.getEngine().getReferenceManager().findCreated();
142            
143            for( Iterator i = pages.iterator(); i.hasNext(); )
144            {
145                String pageName = (String) i.next();
146    
147                if( excludePtrn.matcher( pageName ).matches() ) continue;
148                if( includePtrn.matcher( pageName ).matches() )
149                {
150                    result.add( pageName );
151                }
152            }
153            
154            return result;
155        }
