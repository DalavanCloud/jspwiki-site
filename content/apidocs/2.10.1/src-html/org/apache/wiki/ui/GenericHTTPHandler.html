019    package org.apache.wiki.ui;
020    
021    import org.apache.wiki.WikiContext;
022    
023    /**
024     *  Provides a generic HTTP handler interface.
025     *  
026     *
027     */
028    public interface GenericHTTPHandler
029    {
030        
031        /**
032         *  Get an identifier for this particular AdminBean.  This id MUST
033         *  conform to URI rules.  The ID must also be unique across all HTTPHandlers.
034         *  
035         *  @return the identifier for the bean
036         */
037        String getId();
038        
039        /**
040         *  Return basic HTML.
041         *  
042         *  @param context
043         *  @return the HTML for the bean
044         */
045        String doGet( WikiContext context );
046        
047        /**
048         *  Handles a POST response.
049         *  @param context
050         *  @return the response string resulting from the POST
051         */
052        String doPost( WikiContext context );
053    }
