038    /**
039     *  Returns the Recent Changes in the wiki being a date-sorted list of page names.
040     *
041     *  <p>Parameters : </p>
042     *  <ul>
043     *  <li><b>since</b> - show changes from the last n days, for example since=5 shows only the pages that were changed in the last five days</li>
044     *  <li><b>format</b> - (full|compact) : if "full", then display a long version with all possible info. If "compact", then be as compact as possible.</li>
045     *  <li><b>timeFormat</b> - the time format to use, the default is "HH:mm:ss"</li>
046     *  <li><b>dateFormat</b> - the date format to use, the default is "dd.MM.yyyy"</li>
047     *  </ul>
048     *  
049     */
050    public class RecentChangesPlugin extends AbstractReferralPlugin implements WikiPlugin
051    {
052        /** Parameter name for the separator format.  Value is <tt>{@value}</tt>. */
053        public static final String PARAM_FORMAT = "format";
054        /** Parameter name for the separator timeFormat.  Value is <tt>{@value}</tt>. */
055        public static final String PARAM_TIME_FORMAT = "timeFormat";
056        /** Parameter name for the separator dateFormat.  Value is <tt>{@value}</tt>. */
057        public static final String PARAM_DATE_FORMAT = "dateFormat";
058    
059        /** How many days we show by default. */
060        private static final int    DEFAULT_DAYS = 100*365;
061        public static final String DEFAULT_TIME_FORMAT ="HH:mm:ss";
062        public static final String DEFAULT_DATE_FORMAT ="dd.MM.yyyy";
063    
064        private static Logger log = Logger.getLogger( RecentChangesPlugin.class );
065    
066        private boolean isSameDay( Date a, Date b )
067        {
068            Calendar aa = Calendar.getInstance(); aa.setTime(a);
069            Calendar bb = Calendar.getInstance(); bb.setTime(b);
070    
071            return aa.get( Calendar.YEAR ) == bb.get( Calendar.YEAR ) &&
072                    aa.get( Calendar.DAY_OF_YEAR ) == bb.get( Calendar.DAY_OF_YEAR );
073        }
074    
075        /**
076         * {@inheritDoc}
077         */
078        public String execute( WikiContext context, Map<String, String> params )
079            throws PluginException
080        {
081            int since = TextUtil.parseIntParameter( params.get( "since" ), DEFAULT_DAYS );
082            int      spacing  = 4;
083            boolean  showAuthor = true;
084            boolean  showChangenote = true;
085            int      tablewidth = 4;
086            
087            WikiEngine engine = context.getEngine();
088    
089            //
090            //  Which format we want to see?
091            //
092            if( "compact".equals( params.get(PARAM_FORMAT) ) )
093            {
094                spacing  = 0;
095                showAuthor = false;
096                showChangenote = false;
097                tablewidth = 2;
098            }
099    
100            Calendar sincedate = new GregorianCalendar();
101            sincedate.add( Calendar.DAY_OF_MONTH, -since );
102    
103            log.debug("Calculating recent changes from "+sincedate.getTime());
104    
105            // FIXME: Should really have a since date on the getRecentChanges method.
106            Collection   changes = engine.getRecentChanges();
107            super.initialize( context, params );
108            changes = super.filterCollection( changes );
109            
110            if( changes != null )
111            {
112                Date olddate   = new Date(0);
113    
114                DateFormat fmt = getDateFormat( context, params );
115                DateFormat tfmt = getTimeFormat( context, params );
116    
117                table rt = new table();
118                rt.setCellPadding(spacing).setClass("recentchanges");
119    
120                for( Iterator i = changes.iterator(); i.hasNext(); )
121                {
122                    WikiPage pageref = (WikiPage) i.next();
123    
124                    Date lastmod = pageref.getLastModified();
125    
126                    if( lastmod.before( sincedate.getTime() ) )
127                    {
128                        break;
129                    }
130                    
131                    if( !isSameDay( lastmod, olddate ) )
132                    {
133                        tr row = new tr();
134                        td col = new td();
135                        
136                        col.setColSpan(tablewidth).setClass("date"); 
137                        col.addElement( new b().addElement(fmt.format(lastmod)) );
138    
139                        rt.addElement(row);
140                        row.addElement(col);                    
141                        olddate = lastmod;
142                    }
143    
144                    String link = context.getURL( pageref instanceof Attachment ? WikiContext.ATTACH : WikiContext.VIEW, 
145                                                  pageref.getName() ) ;
146                    
147                    a linkel = new a(link,engine.beautifyTitle(pageref.getName()));
148                    
149                    tr row = new tr();
150                    
151                    td col = new td().setWidth("30%").addElement(linkel);
152    
153                    //
154                    //  Add the direct link to the attachment info.
155                    //
156                    if( pageref instanceof Attachment )
157                    {
158                        linkel = new a().setHref(context.getURL(WikiContext.INFO,pageref.getName()));
159                        linkel.setClass("infolink");
160                        linkel.addElement( new img().setSrc(context.getURL(WikiContext.NONE, "images/attachment_small.png")));
161    
162                        col.addElement( linkel );
163                    }
164    
165                    
166                    row.addElement(col);
167                    rt.addElement(row);
168                    
169                    if( pageref instanceof Attachment )
170                    {
171                        row.addElement( new td(tfmt.format(lastmod)).setClass("lastchange") );
172                    }
173                    else
174                    {
175                        td infocol = (td) new td().setClass("lastchange");
176                        infocol.addElement( new a(context.getURL(WikiContext.DIFF, pageref.getName(), "r1=-1"),tfmt.format(lastmod)) );
177                        row.addElement(infocol);
178                    }
179    
180                    //
181                    //  Display author information.
182                    //
183    
184                    if( showAuthor )
185                    {
186                        String author = pageref.getAuthor();
187    
188                        td authorinfo = new td();
189                        authorinfo.setClass("author");
190                        
191                        if( author != null )
192                        {
193                            if( engine.pageExists(author) )
194                            {
195                                authorinfo.addElement( new a(context.getURL(WikiContext.VIEW, author),author) );
196                            }
197                            else
198                            {
199                                authorinfo.addElement(author);
200                            }
201                        }
202                        else
203                        {
204                            authorinfo.addElement( Preferences.getBundle( context, InternationalizationManager.CORE_BUNDLE )
205                                                              .getString( "common.unknownauthor" ) );
206                        }
207    
208                        row.addElement( authorinfo );
209                    }
210    
211                    // Change note
212                    if( showChangenote )
213                    {
214                        String changenote = (String)pageref.getAttribute(WikiPage.CHANGENOTE);
215                        
216                        row.addElement( new td(changenote != null ? changenote : "").setClass("changenote") );
217                    }
218                    
219                    //  Revert note
220    /*                
221                    if( context.hasAdminPermissions() )
222                    {
223                        row.addElement( new td("Revert") );
224                    }
225     */
226                }
227    
228                rt.setPrettyPrint(true);
229                return rt.toString();
230            }
231            
232            return "";
233        }

237        // TODO: Ideally the default behavior should be to return the default format for the default
238        // locale, but that is at odds with the 1st version of this plugin. We seek to preserve the
239        // behaviour of that first version, so to get the default format, the user must explicitly do
240        // something like: dateFormat='' timeformat='' which is a odd, but probably okay.
241        private DateFormat getTimeFormat( WikiContext context, Map<String, String> params )
242        {
243            String formatString = get(params, DEFAULT_TIME_FORMAT, PARAM_TIME_FORMAT);
244    
245            if ("".equals(formatString.trim()))
246                return Preferences.getDateFormat( context, TimeFormat.TIME );
247    
248            return new SimpleDateFormat(formatString);
249        }
250    
251    
252    
253        private DateFormat getDateFormat( WikiContext context, Map<String, String> params )
254        {
255            String formatString = get(params, DEFAULT_DATE_FORMAT, PARAM_DATE_FORMAT);
256    
257            if( "".equals(formatString.trim() ) )
258            {
259                return Preferences.getDateFormat( context, TimeFormat.DATE );
260            }
261    
262            return new SimpleDateFormat( formatString );
263    
264        }
265    
266    
267    
268        private String get(Map<String, String> params, String defaultValue, String paramName)
269        {
270            String value = params.get(paramName);
271            return null == value ? defaultValue : value;
272        }
