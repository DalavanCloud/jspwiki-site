044    /**
045     * Stores the contents of a plugin in a WikiDocument DOM tree.
046     * <p/>
047     * If the RenderingManager.WYSIWYG_EDITOR_MODE is set to Boolean.TRUE in the context,
048     * then the plugin
049     * is rendered as WikiMarkup.  This allows an HTML editor to work without
050     * rendering the plugin each time as well.
051     * <p/>
052     * If RenderingManager.VAR_EXECUTE_PLUGINS is set to Boolean.FALSE, then
053     * the plugin is not executed.
054     *
055     * @since 2.4
056     */
057    public class PluginContent extends Text {
058    
059        private static final String BLANK = "";
060        private static final String CMDLINE = "_cmdline";
061        private static final String ELEMENT_BR = "<br/>";
062        private static final String EMITTABLE_PLUGINS = "Image|FormOpen|FormClose|FormInput|FormTextarea|FormSelect";
063        private static final String LINEBREAK = "\n";
064        private static final String PLUGIN_START = "[{";
065        private static final String PLUGIN_END = "}]";
066        private static final String SPACE = " ";
067    
068        private static final long serialVersionUID = 1L;
069    
070        private static Logger log = Logger.getLogger(PluginContent.class);
071    
072        private String m_pluginName;
073        private Map<String, String> m_params;
074    
075        /**
076         * Creates a new DOM element with the given plugin name and a map of parameters.
077         *
078         * @param pluginName The FQN of a plugin.
079         * @param parameters A Map of parameters.
080         */
081        public PluginContent(String pluginName, Map<String, String> parameters) {
082            m_pluginName = pluginName;
083            m_params = parameters;
084        }
085    
086        /**
087         * Returns the name of the plugin invoked by the DOM element.
088         *
089         * @return Name of the plugin
090         * @since 2.5.7
091         */
092        public String getPluginName() {
093            return m_pluginName;
094        }
095    
096        /**
097         * Returns a parameter value from the parameter map.
098         *
099         * @param name the name of the parameter.
100         * @return The value from the map, or null, if no such parameter exists.
101         */
102        public String getParameter(String name) {
103            return m_params.get(name);
104        }
105    
106        /**
107         * Returns the parameter map given in the constructor.
108         *
109         * @return The parameter map.
110         */
111        public Map<String, String> getParameters() {
112            return m_params;
113        }
114    
115        /**
116         * Returns the rendered plugin.  Only calls getText().
117         *
118         * @return HTML
119         */
120        public String getValue() {
121            return getText();
122        }
123    
124        /**
125         * The main invocation for the plugin.  When the getText() is called, it
126         * invokes the plugin and returns its contents.  If there is no Document
127         * yet, only returns the plugin name itself.
128         *
129         * @return The plugin rendered according to the options set in the WikiContext.
130         */
131        public String getText() {
132            String result;
133    
134            WikiDocument doc = (WikiDocument) getDocument();
135    
136            if (doc == null) {
137                //
138                // This element has not yet been attached anywhere, so we simply assume there is
139                // no rendering and return the plugin name.  This is required e.g. when the 
140                // paragraphify() checks whether the element is empty or not.  We can't of course
141                // know whether the rendering would result in an empty string or not, but let us
142                // assume it does not.
143                //
144    
145                return getPluginName();
146            }
147    
148            WikiContext context = doc.getContext();
149    
150            Boolean wysiwygVariable = (Boolean) context.getVariable(RenderingManager.WYSIWYG_EDITOR_MODE);
151            boolean wysiwygEditorMode = false;
152            if (wysiwygVariable != null) {
153                wysiwygEditorMode = wysiwygVariable.booleanValue();
154            }
155    
156            try {
157                //
158                //  Determine whether we should emit the actual code for this plugin or
159                //  whether we should execute it.  For some plugins we always execute it,
160                //  since they can be edited visually.
161                //
162                // FIXME: The plugin name matching should not be done here, but in a per-editor resource
163                if (wysiwygEditorMode && !m_pluginName.matches(EMITTABLE_PLUGINS)) {
164                    result = PLUGIN_START + m_pluginName + SPACE;
165    
166                    // convert newlines to <br> in case the plugin has a body.
167                    String cmdLine = (m_params.get(CMDLINE)).replaceAll(LINEBREAK, ELEMENT_BR);
168    
169                    result = result + cmdLine + PLUGIN_END;
170                } else {
171                    Boolean b = (Boolean) context.getVariable(RenderingManager.VAR_EXECUTE_PLUGINS);
172                    if (b != null && !b.booleanValue()) {
173                        return BLANK;
174                    }
175    
176                    WikiEngine engine = context.getEngine();
177    
178                    Map<String, String> parsedParams = new HashMap<String, String>();
179    
180                    //
181                    //  Parse any variable instances from the string
182                    //
183                    for (Map.Entry<String, String> e : m_params.entrySet()) {
184                        String val = e.getValue();
185                        val = engine.getVariableManager().expandVariables(context, val);
186                        parsedParams.put(e.getKey(), val);
187                    }
188                    PluginManager pm = engine.getPluginManager();
189                    result = pm.execute(context, m_pluginName, parsedParams);
190                }
191            } catch (Exception e) {
192                if (wysiwygEditorMode) {
193                    result = "";
194                } else {
195                    // log.info("Failed to execute plugin",e);
196                    ResourceBundle rb = Preferences.getBundle(context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE);
197                    result = JSPWikiMarkupParser.makeError(
198                            MessageFormat.format(rb.getString("plugin.error.insertionfailed"), e.getMessage())).getText();
199                }
200            }
201    
202    
203            return result;
204        }
205    
206        /**
207         * Executes the executeParse() method.
208         *
209         * @param context The WikiContext
210         * @throws PluginException If something goes wrong.
211         */
212        public void executeParse(WikiContext context) throws PluginException {
213            PluginManager pm = context.getEngine().getPluginManager();
214            if (pm.pluginsEnabled()) {
215                ResourceBundle rb = Preferences.getBundle(context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE);
216                Map<String, String> params = getParameters();
217                WikiPlugin plugin = pm.newWikiPlugin(getPluginName(), rb);
218                try {
219                    if (plugin != null && plugin instanceof ParserStagePlugin) {
220                        ((ParserStagePlugin) plugin).executeParser(this, context, params);
221                    }
222                } catch (ClassCastException e) {
223                    throw new PluginException(MessageFormat.format(rb.getString("plugin.error.notawikiplugin"), getPluginName()), e);
224                }
225            }
226        }
227    
228        /**
229         * Parses a plugin invocation and returns a DOM element.
230         *
231         * @param context     The WikiContext
232         * @param commandline The line to parse
233         * @param pos         The position in the stream parsing.
234         * @return A DOM element
235         * @throws PluginException If plugin invocation is faulty
236         * @since 2.10.0
237         */
238        public static PluginContent parsePluginLine(WikiContext context, String commandline, int pos) throws PluginException {
239            PatternMatcher matcher = new Perl5Matcher();
240    
241            try {
242                PluginManager pm = context.getEngine().getPluginManager();
243                if (matcher.contains(commandline, pm.getPluginPattern())) {
244    
245                    MatchResult res = matcher.getMatch();
246    
247                    String plugin = res.group(2);
248                    String args = commandline.substring(res.endOffset(0),
249                            commandline.length() -
250                                    (commandline.charAt(commandline.length() - 1) == '}' ? 1 : 0));
251                    Map<String, String> arglist = pm.parseArgs(args);
252    
253                    // set wikitext bounds of plugin as '_bounds' parameter, e.g., [345,396]
254                    if (pos != -1) {
255                        int end = pos + commandline.length() + 2;
256                        String bounds = pos + "|" + end;
257                        arglist.put(PluginManager.PARAM_BOUNDS, bounds);
258                    }
259    
260                    PluginContent result = new PluginContent(plugin, arglist);
261    
262                    return result;
263                }
264            } catch (ClassCastException e) {
265                log.error("Invalid type offered in parsing plugin arguments.", e);
266                throw new InternalWikiException("Oops, someone offered !String!");
267            } catch (NoSuchElementException e) {
268                String msg = "Missing parameter in plugin definition: " + commandline;
269                log.warn(msg, e);
270                throw new PluginException(msg);
271            } catch (IOException e) {
272                String msg = "Zyrf.  Problems with parsing arguments: " + commandline;
273                log.warn(msg, e);
274                throw new PluginException(msg);
275            }
276    
277            return null;
278        }
