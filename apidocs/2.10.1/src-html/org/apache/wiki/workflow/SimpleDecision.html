019    package org.apache.wiki.workflow;
020    
021    import java.security.Principal;
022    
023    /**
024     * Decision subclass that includes two available Outcomes:
025     * {@link Outcome#DECISION_APPROVE} or {@link Outcome#DECISION_DENY}.
026     * The Decision is reassignable, and the default Outcome is 
027     * {@link Outcome#DECISION_APPROVE}.
028     * 
029     */
030    public class SimpleDecision extends Decision
031    {
032    
033        private static final long serialVersionUID = 8192213077644617341L;
034    
035        /**
036         * Constructs a new SimpleDecision assigned to a specified actor.
037         * @param workflow the parent Workflow
038         * @param messageKey the message key that describes the Decision, which
039         * will be presented in the UI
040         * @param actor the Principal (<em>e.g.</em>, WikiPrincipal,
041         * GroupPrincipal, Role) who will decide
042         */
043        public SimpleDecision( Workflow workflow, String messageKey, Principal actor )
044        {
045            super( workflow, messageKey, actor, Outcome.DECISION_APPROVE );
046    
047            // Add the other default outcomes
048            super.addSuccessor( Outcome.DECISION_DENY, null );
049        }
050    
051    }
