046    /**
047     *  Provides a simple directory based repository for Wiki pages.
048     *  Pages are held in a directory structure:
049     *  <PRE>
050     *    Main.txt
051     *    Foobar.txt
052     *    OLD/
053     *       Main/
054     *          1.txt
055     *          2.txt
056     *          page.properties
057     *       Foobar/
058     *          page.properties
059     *  </PRE>
060     *
061     *  In this case, "Main" has three versions, and "Foobar" just one version.
062     *  <P>
063     *  The properties file contains the necessary metainformation (such as author)
064     *  information of the page.  DO NOT MESS WITH IT!
065     *
066     *  <P>
067     *  All files have ".txt" appended to make life easier for those
068     *  who insist on using Windows or other software which makes assumptions
069     *  on the files contents based on its name.
070     *
071     */
072    public class VersioningFileProvider
073        extends AbstractFileProvider

085        /**
086         *  {@inheritDoc}
087         */
088        public void initialize( WikiEngine engine, Properties properties )
089            throws NoRequiredPropertyException,
090                   IOException
091        {
092            super.initialize( engine, properties );
093            // some additional sanity checks :
094            File oldpages = new File(getPageDirectory(), PAGEDIR);
095            if (!oldpages.exists())
096            {
097                if (!oldpages.mkdirs())
098                {
099                    throw new IOException("Failed to create page version directory " + oldpages.getAbsolutePath());
100                }
101            }
102            else
103            {
104                if (!oldpages.isDirectory())
105                {
106                    throw new IOException("Page version directory is not a directory: " + oldpages.getAbsolutePath());
107                }
108                if (!oldpages.canWrite())
109                {
110                    throw new IOException("Page version directory is not writable: " + oldpages.getAbsolutePath());
111                }
112            }
113            log.info("Using directory " + oldpages.getAbsolutePath() + " for storing old versions of pages");
114        }

116        /**
117         *  Returns the directory where the old versions of the pages
118         *  are being kept.
119         */
120        private File findOldPageDir( String page )
121        {
122            if( page == null )
123            {
124                throw new InternalWikiException("Page may NOT be null in the provider!");
125            }
126    
127            File oldpages = new File( getPageDirectory(), PAGEDIR );
128    
129            return new File( oldpages, mangleName(page) );
130        }

178        private int findLatestVersion( String page )
179        {
180            int version = -1;
181            
182            try
183            {
184                Properties props = getPageProperties( page );
185                
186                for( Iterator i = props.keySet().iterator(); i.hasNext(); )
187                {
188                    String key = (String)i.next();
189                    
190                    if( key.endsWith(".author") )
191                    {
192                        int cutpoint = key.indexOf('.');
193                        if( cutpoint > 0 )
194                        {
195                            String pageNum = key.substring(0,cutpoint);
196                            
197                            try
198                            {
199                                int res = Integer.parseInt( pageNum );
200                                
201                                if( res > version )
202                                {
203                                    version = res;
204                                }
205                            }
206                            catch( NumberFormatException e ) {} // It's okay to skip these. 
207                        }
208                    }
209                }
210            }
211            catch( IOException e )
212            {
213                log.error("Unable to figure out latest version - dying...",e);
214            }
215            
216            return version;
217        }

219        /**
220         *  Reads page properties from the file system.
221         */
222        private Properties getPageProperties( String page )
223            throws IOException
224        {
225            File propertyFile = new File( findOldPageDir(page), PROPERTYFILE );
226    
227            if( propertyFile.exists() )
228            {
229                long lastModified = propertyFile.lastModified();
230    
231                //
232                //   The profiler showed that when calling the history of a page the propertyfile
233                //   was read just as much times as there were versions of that file. The loading
234                //   of a propertyfile is a cpu-intensive job. So now hold on to the last propertyfile
235                //   read because the next method will with a high probability ask for the same propertyfile.
236                //   The time it took to show a historypage with 267 versions dropped with 300%. 
237                //
238                
239                CachedProperties cp = m_cachedProperties;
240                
241                if( cp != null 
242                    && cp.m_page.equals(page) 
243                    && cp.m_lastModified == lastModified)
244                {
245                    return cp.m_props;
246                }
247                
248                InputStream in = null;
249                
250                try
251                {
252                    in = new BufferedInputStream(new FileInputStream( propertyFile ));
253    
254                    Properties props = new Properties();
255    
256                    props.load(in);
257    
258                    cp = new CachedProperties();
259                    cp.m_page = page;
260                    cp.m_lastModified = lastModified;
261                    cp.m_props = props;
262                    
263                    m_cachedProperties = cp; // Atomic
264                    
265                    return props;
266                }
267                finally
268                {
269                    IOUtils.closeQuietly( in );
270                }
271            }
272            
273            return new Properties(); // Returns an empty object
274        }

276        /**
277         *  Writes the page properties back to the file system.
278         *  Note that it WILL overwrite any previous properties.
279         */
280        private void putPageProperties( String page, Properties properties )
281            throws IOException
282        {
283            File propertyFile = new File( findOldPageDir(page), PROPERTYFILE );
284            OutputStream out = null;
285            
286            try
287            {
288                out = new FileOutputStream( propertyFile );
289    
290                properties.store( out, " JSPWiki page properties for "+page+". DO NOT MODIFY!" );
291            }
292            finally
293            {
294                IOUtils.closeQuietly( out );
295            }
296        }

298        /**
299         *  Figures out the real version number of the page and also checks
300         *  for its existence.
301         *
302         *  @throws NoSuchVersionException if there is no such version.
303         */
304        private int realVersion( String page, int requestedVersion )
305            throws NoSuchVersionException,
306                   ProviderException
307        {
308            //
309            //  Quickly check for the most common case.
310            //
311            if( requestedVersion == WikiProvider.LATEST_VERSION )
312            {
313                return -1;
314            }
315    
316            int latest = findLatestVersion(page);
317    
318            if( requestedVersion == latest ||
319                (requestedVersion == 1 && latest == -1 ) )
320            {
321                return -1;
322            }
323            else if( requestedVersion <= 0 || requestedVersion > latest )
324            {
325                throw new NoSuchVersionException("Requested version "+requestedVersion+", but latest is "+latest );
326            }
327    
328            return requestedVersion;
329        }

331        /**
332         *  {@inheritDoc}
333         */
334        public synchronized String getPageText( String page, int version )
335            throws ProviderException
336        {
337            File dir = findOldPageDir( page );
338    
339            version = realVersion( page, version );
340            if( version == -1 )
341            {
342                // We can let the FileSystemProvider take care
343                // of these requests.
344                return super.getPageText( page, WikiPageProvider.LATEST_VERSION );
345            }
346    
347            File pageFile = new File( dir, ""+version+FILE_EXT );
348    
349            if( !pageFile.exists() )
350                throw new NoSuchVersionException("Version "+version+"does not exist.");
351            
352            return readFile( pageFile );
353        }

356        // FIXME: Should this really be here?
357        private String readFile( File pagedata )
358            throws ProviderException
359        {
360            String      result = null;
361            InputStream in     = null;
362    
363            if( pagedata.exists() )
364            {
365                if( pagedata.canRead() )
366                {
367                    try
368                    {          
369                        in = new FileInputStream( pagedata );
370                        result = FileUtil.readContents( in, m_encoding );
371                    }
372                    catch( IOException e )
373                    {
374                        log.error("Failed to read", e);
375                        throw new ProviderException("I/O error: "+e.getMessage());
376                    }
377                    finally
378                    {
379                        IOUtils.closeQuietly( in );
380                    }
381                }
382                else
383                {
384                    log.warn("Failed to read page from '"+pagedata.getAbsolutePath()+"', possibly a permissions problem");
385                    throw new ProviderException("I cannot read the requested page.");
386                }
387            }
388            else
389            {
390                // This is okay.
391                // FIXME: is it?
392                log.info("New page");
393            }
394    
395            return result;
396        }

400        /*
401          This is how the page directory should look like:
402    
403             version    pagedir       olddir
404              none       empty         empty
405               1         Main.txt (1)  empty
406               2         Main.txt (2)  1.txt
407               3         Main.txt (3)  1.txt, 2.txt
408        */
409        /**
410         *  {@inheritDoc}
411         */
412        public synchronized void putPageText( WikiPage page, String text )
413            throws ProviderException
414        {
415            //
416            //  This is a bit complicated.  We'll first need to
417            //  copy the old file to be the newest file.
418            //
419    
420            File pageDir = findOldPageDir( page.getName() );
421    
422            if( !pageDir.exists() )
423            {
424                pageDir.mkdirs();
425            }
426    
427            int  latest  = findLatestVersion( page.getName() );
428    
429            try
430            {
431                //
432                // Copy old data to safety, if one exists.
433                //
434    
435                File oldFile = findPage( page.getName() );
436    
437                // Figure out which version should the old page be?
438                // Numbers should always start at 1.
439                // "most recent" = -1 ==> 1
440                // "first"       = 1  ==> 2
441    
442                int versionNumber = (latest > 0) ? latest : 1;
443    
444                if( oldFile != null && oldFile.exists() )
445                {
446                    InputStream in = null;
447                    OutputStream out = null;
448                    
449                    try
450                    {
451                        in = new BufferedInputStream( new FileInputStream( oldFile ) );
452                        File pageFile = new File( pageDir, Integer.toString( versionNumber )+FILE_EXT );
453                        out = new BufferedOutputStream( new FileOutputStream( pageFile ) );
454    
455                        FileUtil.copyContents( in, out );
456    
457                        //
458                        // We need also to set the date, since we rely on this.
459                        //
460                        pageFile.setLastModified( oldFile.lastModified() );
461    
462                        //
463                        // Kludge to make the property code to work properly.
464                        //
465                        versionNumber++;
466                    }
467                    finally
468                    {
469                        IOUtils.closeQuietly( out );
470                        IOUtils.closeQuietly( in );
471                    }
472                }
473    
474                //
475                //  Let superclass handler writing data to a new version.
476                //
477    
478                super.putPageText( page, text );
479    
480                //
481                //  Finally, write page version data.
482                //
483    
484                // FIXME: No rollback available.
485                Properties props = getPageProperties( page.getName() );
486    
487                props.setProperty( versionNumber+".author", (page.getAuthor() != null) ? page.getAuthor() : "unknown" );
488                
489                String changeNote = (String) page.getAttribute(WikiPage.CHANGENOTE);
490                if( changeNote != null )
491                {
492                    props.setProperty( versionNumber+".changenote", changeNote );
493                }
494                
495                putPageProperties( page.getName(), props );
496            }
497            catch( IOException e )
498            {
499                log.error( "Saving failed", e );
500                throw new ProviderException("Could not save page text: "+e.getMessage());
501            }
502        }

504        /**
505         *  {@inheritDoc}
506         */
507        public WikiPage getPageInfo( String page, int version )
508            throws ProviderException
509        {
510            int latest = findLatestVersion(page);
511            int realVersion;
512    
513            WikiPage p = null;
514    
515            if( version == WikiPageProvider.LATEST_VERSION ||
516                version == latest || 
517                (version == 1 && latest == -1) )
518            {
519                //
520                // Yes, we need to talk to the top level directory
521                // to get this version.
522                //
523                // I am listening to Press Play On Tape's guitar version of
524                // the good old C64 "Wizardry" -tune at this moment.
525                // Oh, the memories...
526                //
527                realVersion = (latest >= 0) ? latest : 1;
528    
529                p = super.getPageInfo( page, WikiPageProvider.LATEST_VERSION );
530    
531                if( p != null )
532                {
533                    p.setVersion( realVersion );
534                }
535            }
536            else
537            {
538                //
539                //  The file is not the most recent, so we'll need to
540                //  find it from the deep trenches of the "OLD" directory
541                //  structure.
542                //
543                realVersion = version;
544                File dir = findOldPageDir( page );
545    
546                if( !dir.exists() || !dir.isDirectory() )
547                {
548                    return null;
549                }
550    
551                File file = new File( dir, version+FILE_EXT );
552    
553                if( file.exists() )
554                {
555                    p = new WikiPage( m_engine, page );
556    
557                    p.setLastModified( new Date(file.lastModified()) );
558                    p.setVersion( version );
559                }
560            }
561    
562            //
563            //  Get author and other metadata information
564            //  (Modification date has already been set.)
565            //
566            if( p != null )
567            {
568                try
569                {
570                    Properties props = getPageProperties( page );
571                    String author = props.getProperty( realVersion+".author" );
572                    if( author != null )
573                    {
574                        p.setAuthor( author );
575                    }
576                    
577                    String changenote = props.getProperty( realVersion+".changenote" );
578                    if( changenote != null ) p.setAttribute( WikiPage.CHANGENOTE, changenote );
579                    
580                }
581                catch( IOException e )
582                {
583                    log.error( "Cannot get author for page"+page+": ", e );
584                }
585            }
586    
587            return p;
588        }

590        /**
591         *  {@inheritDoc}
592         */
593        public boolean pageExists( String pageName, int version )
594        {
595            if (version == WikiPageProvider.LATEST_VERSION || version == findLatestVersion( pageName ) ) {
596                return pageExists(pageName);
597            }
598    
599            File dir = findOldPageDir( pageName );
600    
601            if( !dir.exists() || !dir.isDirectory() )
602            {
603                return false;
604            }
605    
606            File file = new File( dir, version+FILE_EXT );
607    
608            return file.exists();
609    
610        }

638        /**
639         *  Removes the relevant page directory under "OLD" -directory as well,
640         *  but does not remove any extra subdirectories from it.  It will only
641         *  touch those files that it thinks to be WikiPages.
642         *  
643         *  @param page {@inheritDoc}
644         *  @throws {@inheritDoc}
645         */
646        // FIXME: Should log errors.
647        public void deletePage( String page )
648            throws ProviderException
649        {
650            super.deletePage( page );
651    
652            File dir = findOldPageDir( page );
653    
654            if( dir.exists() && dir.isDirectory() )
655            {
656                File[] files = dir.listFiles( new WikiFileFilter() );
657    
658                for( int i = 0; i < files.length; i++ )
659                {
660                    files[i].delete();
661                }
662    
663                File propfile = new File( dir, PROPERTYFILE );
664    
665                if( propfile.exists() )
666                {
667                    propfile.delete();
668                }
669    
670                dir.delete();
671            }
672        }

674        /**
675         *  {@inheritDoc}
676         *  
677         *  Deleting versions has never really worked,
678         *  JSPWiki assumes that version histories are "not gappy". 
679         *  Using deleteVersion() is definitely not recommended.
680         *  
681         */
682        public void deleteVersion( String page, int version )
683            throws ProviderException
684        {
685            File dir = findOldPageDir( page );
686    
687            int latest = findLatestVersion( page );
688    
689            if( version == WikiPageProvider.LATEST_VERSION ||
690                version == latest || 
691                (version == 1 && latest == -1) )
692            {
693                //
694                //  Delete the properties
695                //
696                try
697                {
698                    Properties props = getPageProperties( page );
699                    props.remove( ((latest > 0) ? latest : 1)+".author" );
700                    putPageProperties( page, props );
701                }
702                catch( IOException e )
703                {
704                    log.error("Unable to modify page properties",e);
705                    throw new ProviderException("Could not modify page properties: " + e.getMessage());
706                }
707    
708                // We can let the FileSystemProvider take care
709                // of the actual deletion
710                super.deleteVersion( page, WikiPageProvider.LATEST_VERSION );
711                
712                //
713                //  Copy the old file to the new location
714                //
715                latest = findLatestVersion( page );
716                
717                File pageDir = findOldPageDir( page );
718                File previousFile = new File( pageDir, Integer.toString(latest)+FILE_EXT );
719    
720                InputStream in = null;
721                OutputStream out = null;
722                
723                try
724                {
725                    if( previousFile.exists() )
726                    {
727                        in = new BufferedInputStream( new FileInputStream( previousFile ) );
728                        File pageFile = findPage(page);
729                        out = new BufferedOutputStream( new FileOutputStream( pageFile ) );
730    
731                        FileUtil.copyContents( in, out );
732    
733                        //
734                        // We need also to set the date, since we rely on this.
735                        //
736                        pageFile.setLastModified( previousFile.lastModified() );
737                    }
738                }
739                catch( IOException e )
740                {
741                    log.fatal("Something wrong with the page directory - you may have just lost data!",e);
742                }
743                finally
744                {
745                    IOUtils.closeQuietly( in );
746                    IOUtils.closeQuietly( out );
747                }
748                
749                return;
750            }
751    
752            File pageFile = new File( dir, ""+version+FILE_EXT );
753    
754            if( pageFile.exists() )
755            {
756                if( !pageFile.delete() )
757                {
758                    log.error("Unable to delete page.");
759                }
760            }
761            else
762            {
763                throw new NoSuchVersionException("Page "+page+", version="+version);
764            }
765        }

775

796        /**
797         *  {@inheritDoc}
798         */
799        public void movePage( String from,
800                              String to )
801            throws ProviderException
802        {
803            // Move the file itself
804            File fromFile = findPage( from );
805            File toFile = findPage( to );
806            
807            fromFile.renameTo( toFile );
808            
809            // Move any old versions
810            File fromOldDir = findOldPageDir( from );
811            File toOldDir = findOldPageDir( to );
812            
813            fromOldDir.renameTo( toOldDir );
814        }
